    exit_code = 2

        super().__init__(message)
        self.ctx = ctx

        if self.ctx is not None:

import textwrap


class HelpFormatter:
    def __init__(self, indent_increment=2, max_width=None, width=None):
        self.indent_increment = indent_increment
        self.width = width or max_width or 80
        self.current_indent = 0
        self.buffer = []

    def write(self, string):
        self.buffer.append(string)

    def indent(self):
        self.current_indent += self.indent_increment

    def dedent(self):
        self.current_indent -= self.indent_increment

    def write_usage(self, prog, args="", prefix="Usage: "):
        self.write(f"{prefix}{prog} {args}\n")

    def write_paragraph(self):
        if self.buffer:
            self.write("\n")

    def write_text(self, text):
        self.write(textwrap.fill(text, self.width - self.current_indent) + "\n")

    def write_dl(self, rows, short_limit=0.5, col_max=30, col_spacing=2):
        for first, second in rows:
            self.write(f"{'':>{self.current_indent}}{first:<{col_max}}{second}\n")

    def write_heading(self, heading):
        self.write(f"{'':>{self.current_indent}}{heading}:\n")

    def section(self, name):
        return _Section(self, name)

    def getvalue(self):
        return "".join(self.buffer)


class _Section:
    def __init__(self, formatter, name):

    def __enter__(self):
        self.formatter.write_heading(self.name)

        return self

    def __exit__(self, *args):
        self.formatter.dedent()
        self.formatter.write_paragraph()

class StringParamType(ParamType):
    name = "TEXT"

    def convert(self, value, param, ctx):
        if isinstance(value, bytes):
            try:
                value = value.decode(ctx.obj.get("encoding", "utf-8") if ctx and ctx.obj else "utf-8")
            except UnicodeDecodeError:
                value = value.decode("latin-1")
        return value

class BoolParamType(ParamType):
    name = "BOOLEAN"

    def convert(self, value, param, ctx):
        if isinstance(value, bool):
            return value
        value = value.lower()
        if value in ("1", "true", "t", "yes", "y", "on"):
            return True
        elif value in ("0", "false", "f", "no", "n", "off"):
            return False
        self.fail(f"{value!r} is not a valid boolean.", param, ctx)

        self.obj = kwargs.get("obj", None)
        self._close_callbacks = []

        push_context(self)
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):

    def close(self):

        return f

    def fail(self, message):
        raise UsageError(message, ctx=self)

    def make_formatter(self):
        return HelpFormatter(max_width=self.max_content_width)


class BaseCommand:
    def __init__(self, name, callback=None, params=None, help=None,
                 epilog=None, short_help=None, add_help_option=True,
                 no_args_is_help=False, hidden=False, deprecated=False):
        self.name = name
        self.callback = callback
        self.params = params or []
        self.help = help
        self.epilog = epilog
        self.short_help = short_help
        self.add_help_option = add_help_option
        self.no_args_is_help = no_args_is_help
        self.hidden = hidden
        self.deprecated = deprecated

    def get_help(self, ctx):
        formatter = ctx.make_formatter()
        self.format_help(ctx, formatter)
        return formatter.getvalue()

    def format_help(self, ctx, formatter):
        self.format_usage(ctx, formatter)
        self.format_help_text(ctx, formatter)
        self.format_options(ctx, formatter)
        self.format_epilog(ctx, formatter)

    def format_usage(self, ctx, formatter):
        formatter.write_usage(ctx.info_name or self.name or "")

    def format_help_text(self, ctx, formatter):
        if self.help:
            formatter.write_paragraph()
            formatter.write_text(self.help)

    def format_options(self, ctx, formatter):
        pass

    def format_epilog(self, ctx, formatter):
        if self.epilog:
            formatter.write_paragraph()
            formatter.write_text(self.epilog)

    def make_context(self, info_name, args, parent=None, **extra):
        ctx = Context(self, parent=parent, info_name=info_name, **extra)
        return ctx

    def parse_args(self, ctx, args):
        return args

    def invoke(self, ctx):
        if self.callback is not None:
            return ctx.invoke(self.callback, **ctx.params)

    def main(self, args=None, prog_name=None, complete_var=None,
             standalone_mode=True, **extra):
        if args is None:
            args = sys.argv[1:]
        else:
            args = list(args)

        prog_name = prog_name or os.path.basename(sys.argv[0])

        try:
            try:
                with self.make_context(prog_name, args, **extra) as ctx:
                    rv = self.invoke(ctx)
                    if standalone_mode:
                        sys.exit(0)
                    return rv
            except (EOFError, KeyboardInterrupt):
                if standalone_mode:
                    sys.exit(1)
                raise Abort()
        except Abort:
            if standalone_mode:
                sys.stderr.write("Aborted!\n")
                sys.exit(1)
            raise
        except Exit as e:
            if standalone_mode:
                sys.exit(e.exit_code)
            return e.exit_code
        except ClickException as e:
            if standalone_mode:
                e.show()
                sys.exit(e.exit_code)
            raise

    def __call__(self, *args, **kwargs):
        return self.main(*args, **kwargs)


class Command(BaseCommand):
    def __init__(self, name, callback=None, params=None, help=None,
                 epilog=None, short_help=None, add_help_option=True,
                 no_args_is_help=False, hidden=False, deprecated=False):
        super().__init__(name, callback=callback, params=params, help=help,
                         epilog=epilog, short_help=short_help,
                         add_help_option=add_help_option,
                         no_args_is_help=no_args_is_help,
                         hidden=hidden, deprecated=deprecated)
        if add_help_option:
            self.params.append(HelpOption(["-h", "--help"]))

    def format_options(self, ctx, formatter):
        opts = []
        for param in self.params:
            rv = param.get_help_record(ctx)
            if rv is not None:
                opts.append(rv)
        if opts:
            with formatter.section("Options"):
                formatter.write_dl(opts)

    def parse_args(self, ctx, args):
        parser = self._make_parser(ctx)
        opts, args, param_order = parser.parse_args(args=args)
        for param in self.params:
            value, args = param.consume_value(ctx, opts)
            param.full_process_value(ctx, value)
        ctx.args = args
        return args

    def _make_parser(self, ctx):
        from .parser import OptionParser
        parser = OptionParser(ctx)
        for param in self.params:
            param.add_to_parser(parser, ctx)
        return parser

    def invoke(self, ctx):
        if self.callback is not None:
            return ctx.invoke(self.callback, **ctx.params)

    def make_context(self, info_name, args, parent=None, **extra):
        ctx = Context(self, parent=parent, info_name=info_name, **extra)
        with ctx:
            self.parse_args(ctx, args)
        # Re-enter for the invoke phase
        push_context(ctx)
        return ctx


class MultiCommand(Command):
    def __init__(self, name=None, invoke_without_command=False,
                 no_args_is_help=None, subcommand_metavar=None,
                 chain=False, result_callback=None, **attrs):
        super().__init__(name, **attrs)
        self.invoke_without_command = invoke_without_command
        self.subcommand_metavar = subcommand_metavar
        self.chain = chain
        self.result_callback = result_callback
        if no_args_is_help is None:
            self.no_args_is_help = not invoke_without_command
        else:
            self.no_args_is_help = no_args_is_help

    def list_commands(self, ctx):
        return []

    def get_command(self, ctx, cmd_name):
        return None

    def invoke(self, ctx):
        def _process_result(value):
            if self.result_callback is not None:
                value = ctx.invoke(self.result_callback, value, **ctx.params)
            return value

        with ctx:
            if not ctx.args and self.invoke_without_command:
                if self.callback is not None:
                    ctx.invoke(self.callback, **ctx.params)
                return _process_result([])

            args = ctx.protected_args + ctx.args
            ctx.args = []
            ctx.protected_args = []

            cmd_name, cmd, args = self.resolve_command(ctx, args)
            if cmd is None:
                return _process_result([])

            ctx.invoked_subcommand = cmd_name
            with cmd.make_context(cmd_name, args, parent=ctx) as sub_ctx:
                rv = [sub_ctx.command.invoke(sub_ctx)]
                return _process_result(rv)

    def resolve_command(self, ctx, args):
        cmd_name = args[0] if args else None
        if cmd_name is None:
            return None, None, args
        cmd = self.get_command(ctx, cmd_name)
        return cmd_name, cmd, args[1:]

class Option:
    def __init__(self, param_decls, default=None, required=False,
                 type=None, help=None, is_flag=False, flag_value=None,
                 multiple=False, count=False, allow_from_autoenv=False,
                 prompt=False, confirmation_prompt=False, hide_input=False,
                 is_eager=False, expose_value=True, callback=None,
                 envvar=None, show_default=False, nargs=1):
        self.param_decls = param_decls
        self.name = self._parse_name(param_decls)
        self.default = default
        self.required = required
        self.type = type
        self.help = help
        self.is_flag = is_flag
        self.flag_value = flag_value
        self.multiple = multiple
        self.count = count
        self.is_eager = is_eager
        self.expose_value = expose_value
        self.callback = callback
        self.envvar = envvar
        self.show_default = show_default
        self.nargs = nargs
        self.prompt = prompt

    def _parse_name(self, param_decls):
        for decl in param_decls:
            if decl.startswith("--"):
                return decl[2:].replace("-", "_")
            elif decl.startswith("-") and len(decl) == 2:
                pass
        for decl in param_decls:
            if decl.startswith("-"):
                return decl.lstrip("-").replace("-", "_")
        return param_decls[0]

    def get_help_record(self, ctx):
        decls = ", ".join(self.param_decls)
        help_str = self.help or ""
        return (decls, help_str)

    def add_to_parser(self, parser, ctx):
        parser.add_option(self.param_decls, dest=self.name,
                          is_flag=self.is_flag, default=self.default,
                          nargs=self.nargs, multiple=self.multiple)

    def consume_value(self, ctx, opts):
        value = opts.get(self.name, self.default)
        return value, opts

    def full_process_value(self, ctx, value):
        if value is None:
            value = self.default
        if self.expose_value:
            ctx.params[self.name] = value
        return value

class OptionParser:
    """A simple option parser for Click."""

    def __init__(self, ctx=None):
        self.ctx = ctx
        self._options = {}
        self._arguments = []
        self._option_order = []

    def add_option(self, param_decls, dest, is_flag=False, default=None,
                   nargs=1, multiple=False, is_eager=False,
                   expose_value=True, callback=None):
        option_info = {
            "dest": dest,
            "is_flag": is_flag,
            "default": default,
            "nargs": nargs,
            "multiple": multiple,
            "is_eager": is_eager,
            "expose_value": expose_value,
            "callback": callback,
        }
        for decl in param_decls:
            self._options[decl] = option_info
        self._option_order.append((param_decls, option_info))

    def add_argument(self, name, nargs=1):
        self._arguments.append({"name": name, "nargs": nargs})

    def parse_args(self, args):
        opts = {}
        remaining = []
        param_order = []

        # Set defaults
        for param_decls, info in self._option_order:
            dest = info["dest"]
            if dest not in opts:
                if info["is_flag"]:
                    opts[dest] = info["default"] if info["default"] is not None else False
                elif info["multiple"]:
                    opts[dest] = ()
                else:
                    opts[dest] = info["default"]

        i = 0
        while i < len(args):
            arg = args[i]
            if arg in self._options:
                info = self._options[arg]
                dest = info["dest"]
                if info["is_flag"]:
                    value = True
                    if info.get("callback") and self.ctx:
                        info["callback"](self.ctx, None, value)
                    elif info.get("callback"):
                        info["callback"](None, None, value)
                    if info.get("expose_value", True):
                        opts[dest] = value
                else:
                    nargs = info["nargs"]
                    if nargs == 1:
                        i += 1
                        if i < len(args):
                            value = args[i]
                        else:
                            value = None
                    else:
                        values = []
                        for _ in range(nargs):
                            i += 1
                            if i < len(args):
                                values.append(args[i])
                        value = tuple(values)
                    if info.get("expose_value", True):
                        if info["multiple"]:
                            opts[dest] = opts.get(dest, ()) + (value,)
                        else:
                            opts[dest] = value
                param_order.append(dest)
            elif arg.startswith("-") and len(arg) > 1:
                # Try to find matching option
                found = False
                for key, info in self._options.items():
                    if arg.startswith(key + "="):
                        dest = info["dest"]
                        value = arg[len(key) + 1:]
                        if info.get("expose_value", True):
                            opts[dest] = value
                        found = True
                        break
                if not found:
                    remaining.append(arg)
            else:
                remaining.append(arg)
            i += 1

        # Process arguments
        arg_idx = 0
        for arg_info in self._arguments:
            name = arg_info["name"]
            nargs = arg_info["nargs"]
            if nargs == 1:
                if arg_idx < len(remaining):
                    opts[name] = remaining[arg_idx]
                    arg_idx += 1
            elif nargs == -1:
                opts[name] = tuple(remaining[arg_idx:])
                arg_idx = len(remaining)
            else:
                opts[name] = tuple(remaining[arg_idx:arg_idx + nargs])
                arg_idx += nargs

        return opts, remaining[arg_idx:], param_order

class Result:
    """Holds the captured result of a CLI invocation."""

    def __init__(self, runner, stdout_bytes, stderr_bytes, exit_code, exception, exc_info=None, return_value=None):
        self.runner = runner
        self.stdout_bytes = stdout_bytes
        self.stderr_bytes = stderr_bytes
        self.exit_code = exit_code
        self.exception = exception
        self.exc_info = exc_info
        self.return_value = return_value

    @property
    def output(self):
        return self.stdout

    @property
    def stdout(self):
        return self.stdout_bytes.decode(self.runner.charset, "replace").replace("\r\n", "\n")

    @property
    def stderr(self):
        if not self.runner.mix_stderr:
            return self.stderr_bytes.decode(self.runner.charset, "replace").replace("\r\n", "\n")
        raise ValueError("stderr is mixed with stdout; use output instead")

    def __repr__(self):
        return f"<Result {self.exception!r}>" if self.exception else "<Result okay>"

    def invoke(self, cli, args=None, input=None, env=None, catch_exceptions=True,
               color=False, **extra):

        return_value = None
        exit_code = 0
        exception = None

        if args is None:
            args = []
        elif isinstance(args, str):
            import shlex
            args = shlex.split(args)
        else:
            args = list(args)

        with self.isolation(input=input, env=env, color=color) as outstreams:
            stdout_buf, stderr_buf, stdout_wrapper, stderr_wrapper = outstreams
            try:
                return_value = cli.main(
                    args=args,
                    prog_name=cli.name or "root",
                    standalone_mode=False,
                    **extra,
                )
            except SystemExit as e:
                exc_info = sys.exc_info()
                e_code = e.code
                if e_code is None:
                    e_code = 0
                if e_code != 0:
                    exception = e
                if not isinstance(e_code, int):
                    sys.stdout.write(str(e_code))
                    sys.stdout.write("\n")
                    e_code = 1

                    raise

                exc_info = sys.exc_info()
                exit_code = 1

                    sys.stderr.flush()
                except Exception:

            stderr_bytes = stderr_buf.getvalue()

            runner=self,

"""
Tests for the fix: CliRunner captures unflushed stderr writes.

Regression test for: CliRunner should capture data written to stderr
without an explicit flush (e.g. print(..., flush=False)).
"""
import sys
import click
from click.testing import CliRunner


def test_unflushed_stderr_is_captured():
    """Unflushed writes to stderr must appear in result.stderr."""

    @click.command()
    def cli():
        # Write to stderr WITHOUT explicit flush
        print("error message", file=sys.stderr, flush=False)

    runner = CliRunner(mix_stderr=False)
    result = runner.invoke(cli, [])
    assert result.exit_code == 0
    assert "error message" in result.stderr, (
        f"Expected 'error message' in stderr, got: {result.stderr!r}"
    )


def test_unflushed_stdout_is_captured():
    """Unflushed writes to stdout must appear in result.output."""

    @click.command()
    def cli():
        print("hello world", flush=False)

    runner = CliRunner()
    result = runner.invoke(cli, [])
    assert result.exit_code == 0
    assert "hello world" in result.output, (
        f"Expected 'hello world' in output, got: {result.output!r}"
    )
