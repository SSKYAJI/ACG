+ from click.core import Command
+ from click.core import Group
+ from click.core import Option
+ from click.shell_completion import CompletionItem
+ from click.shell_completion import FishComplete
+ 
+ 
+ def _fish_format(item):
+     comp = FishComplete.__new__(FishComplete)
+     return comp.format_completion(item)
+ 
+ 
+ def test_fish_multiline_help_single_line():
+     item = CompletionItem("value", help="short help")
+     assert _fish_format(item) == "plain,value\tshort help"
+ 
+ 
+ def test_fish_multiline_help_multiline():
+     item = CompletionItem("value", help="first line\nsecond line\nthird line")
+     result = _fish_format(item)
+     assert "\n" not in result
+     assert result == "plain,value\tfirst line"
+ 
+ 
+ def test_fish_multiline_help_no_help():
+     item = CompletionItem("value")
+     assert _fish_format(item) == "plain,value"
+ 
+ 
+ def test_fish_multiline_help_windows_line_endings():
+     item = CompletionItem("value", help="first line\r\nsecond line")
+     result = _fish_format(item)
+     # splitlines() handles \r\n correctly
+     assert "\n" not in result
+     assert "\r" not in result
+     assert "first line" in result
