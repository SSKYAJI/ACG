+ def cli_with_unflushed_stderr(args, standalone_mode=False, **kwargs):
+     """Simulate a CLI that writes to stderr without explicit flush."""
+     # Using print with flush=False (default) - data stays in buffer until flush
+     print("error message", file=sys.stderr, flush=False)
+     print("normal output", file=sys.stdout, flush=False)
+
+
+ def cli_with_multiple_stderr_writes(args, standalone_mode=False, **kwargs):
+     """Simulate a CLI that writes multiple lines to stderr without flushing."""

+
+
+ def cli_stderr_no_newline(args, standalone_mode=False, **kwargs):
