+ import enum

+ # Basic types that can be handled directly
+ BASIC_TYPES = (bool, int, float, str, bytes, type(None))

+ class OptionType(enum.Enum):
+     UNPROCESSED = "unprocessed"
+     STRING = "string"
+     INT = "int"
+     FLOAT = "float"
+     BOOL = "bool"
+     BYTES = "bytes"
+ 
+ 
+ # Mapping from basic Python types to OptionType
+ _TYPE_MAP = {
+     str: OptionType.STRING,
+     int: OptionType.INT,
+     float: OptionType.FLOAT,
+     bool: OptionType.BOOL,
+     bytes: OptionType.BYTES,
+ }
+

+ class Option:
+     """Represents a command-line option.
+ 
+     When flag_value is set to a non-basic type (e.g. an enum value),
+     the option's type is automatically set to UNPROCESSED instead of
+     raising a TypeError or mishandling the value during parsing.
+     """
+ 
+     def __init__(self, name, flag_value=None, option_type=None, default=None,
+                  help=None, metavar=None, multiple=False):
+         """Initialize an Option.
+ 
+         Args:
+             name: The name of the option (e.g. '--verbose').
+             flag_value: The value to use when the flag is present without
+                         an explicit value. If this is a non-basic type,
+                         option_type is automatically set to UNPROCESSED.
+             option_type: The OptionType for this option. If None, it is
+                          inferred from flag_value or defaults to STRING.
+             default: The default value when the option is not present.
+             help: Help text for the option.
+             metavar: Metavar string for usage display.
+             multiple: Whether the option can be specified multiple times.
+         """
+         self.name = name
+         self.flag_value = flag_value
+         self.default = default
+         self.help = help
+         self.metavar = metavar
+         self.multiple = multiple
+ 
+         if option_type is not None:
+             # Explicit type provided — use it directly
+             self.option_type = option_type
+         elif flag_value is not None:
+             # Infer type from flag_value
+             fv_type = type(flag_value)
+             if fv_type in _TYPE_MAP:
+                 self.option_type = _TYPE_MAP[fv_type]
+             else:
+                 # Non-basic type: use UNPROCESSED instead of raising TypeError
+                 self.option_type = OptionType.UNPROCESSED
+         else:
+             self.option_type = OptionType.STRING
+ 
+     def parse(self, value):
+         """Parse a string value according to this option's type.
+ 
+         Args:
+             value: The raw string value from the command line, or None
+                    if the flag was given without a value (in which case
+                    flag_value is returned).
+ 
+         Returns:
+             The parsed value.
+         """
+         if value is None:
+             return self.flag_value
+ 
+         if self.option_type == OptionType.UNPROCESSED:
+             # Return the raw value without conversion
+             return value
+         elif self.option_type == OptionType.STRING:
+             return str(value)
+         elif self.option_type == OptionType.INT:
+             return int(value)
+         elif self.option_type == OptionType.FLOAT:
+             return float(value)
+         elif self.option_type == OptionType.BOOL:
+             if isinstance(value, bool):
+                 return value
+             if isinstance(value, str):
+                 lower = value.lower()
+                 if lower in ('true', '1', 'yes'):
+                     return True
+                 if lower in ('false', '0', 'no'):
+                     return False
+             raise ValueError(f"Cannot convert {value!r} to bool")
+         elif self.option_type == OptionType.BYTES:
+             if isinstance(value, bytes):
+                 return value
+             return value.encode('utf-8')
+         else:
+             return value
