+ // sanitizeFishDescription sanitizes a help string for use in fish shell completion.
+ // Fish completion's `complete` command does not support multiline descriptions,
+ // so we collapse any newlines (and surrounding whitespace) into a single space,
+ // and also escape any special characters that would break the fish `complete` command.
+ func sanitizeFishDescription(s string) string {
+ 	// Split on newlines and rejoin with a space, trimming each segment.
+ 	lines := strings.Split(s, "\n")
+ 	parts := make([]string, 0, len(lines))
+ 	for _, line := range lines {
+ 		trimmed := strings.TrimSpace(line)
+ 		if trimmed != "" {
+ 			parts = append(parts, trimmed)
+ 		}
+ 	}
+ 	result := strings.Join(parts, " ")
+ 
+ 	// Escape single quotes for fish shell safety.
+ 	result = strings.ReplaceAll(result, "'", "\\'")
+ 
+ 	return result
+ }
+ 
+ // fishCompletionCommand returns a fish `complete` command string for a given
+ // command name, flag name, short flag, and description.
+ func fishCompletionCommand(cmdName, longFlag, shortFlag, description string) string {
+ 	desc := sanitizeFishDescription(description)
+ 
+ 	var sb strings.Builder
+ 	sb.WriteString(fmt.Sprintf("complete -c %s", cmdName))
+ 	if shortFlag != "" {
+ 		sb.WriteString(fmt.Sprintf(" -s %s", shortFlag))
+ 	}
+ 	if longFlag != "" {
+ 		sb.WriteString(fmt.Sprintf(" -l %s", longFlag))
+ 	}
+ 	if desc != "" {
+ 		sb.WriteString(fmt.Sprintf(" -d '%s'", desc))
+ 	}
+ 	return sb.String()
+ }
