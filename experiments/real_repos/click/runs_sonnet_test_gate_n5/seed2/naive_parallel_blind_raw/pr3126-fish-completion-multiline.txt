+ // sanitizeForFish removes or replaces characters that would break fish completion scripts.
+ // In particular, newlines in help text must be collapsed to spaces.
+ func sanitizeForFish(s string) string {
+ 	// Replace all newline variants with a space
+ 	s = strings.ReplaceAll(s, "\r\n", " ")
+ 	s = strings.ReplaceAll(s, "\r", " ")
+ 	s = strings.ReplaceAll(s, "\n", " ")
+ 	// Replace single quotes with escaped version safe for fish
+ 	s = strings.ReplaceAll(s, "'", "\\'")
+ 	// Collapse multiple spaces into one
+ 	for strings.Contains(s, "  ") {
+ 		s = strings.ReplaceAll(s, "  ", " ")
+ 	}
+ 	return strings.TrimSpace(s)
+ }
+ 
+ // fishCompletionEntry represents a single fish completion line.
+ type fishCompletionEntry struct {
+ 	Command     string
+ 	Short       string
+ 	Long        string
+ 	Description string
+ 	IsCommand   bool // true if this is a subcommand, false if it's a flag/option
+ }
+ 
+ // render produces the fish complete command string for this entry.
+ func (e fishCompletionEntry) render() string {
+ 	desc := sanitizeForFish(e.Description)
+ 	var sb strings.Builder
+ 	sb.WriteString("complete -c ")
+ 	sb.WriteString(e.Command)
+ 	if e.IsCommand {
+ 		sb.WriteString(" -f -n '__fish_use_subcommand'")
+ 		sb.WriteString(" -a '")
+ 		sb.WriteString(e.Long)
+ 		sb.WriteString("'")
+ 	} else {
+ 		if e.Short != "" {
+ 			sb.WriteString(" -s ")
+ 			sb.WriteString(e.Short)
+ 		}
+ 		if e.Long != "" {
+ 			sb.WriteString(" -l ")
+ 			sb.WriteString(e.Long)
+ 		}
+ 	}
+ 	if desc != "" {
+ 		sb.WriteString(" -d '")
+ 		sb.WriteString(desc)
+ 		sb.WriteString("'")
+ 	}
+ 	return sb.String()
+ }
+ 
+ // GenerateFishCompletions generates fish shell completion script lines
+ // for the given command entries.
+ func GenerateFishCompletions(entries []fishCompletionEntry) string {
+ 	var lines []string
+ 	for _, e := range entries {
+ 		lines = append(lines, e.render())
+ 	}
+ 	return strings.Join(lines, "\n")
+ }
