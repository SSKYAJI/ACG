+"""Tests for Option flag_value with non-basic types."""
+import enum
+import pytest
+
+
+class Color(enum.Enum):
+    RED = 1
+    GREEN = 2
+    BLUE = 3
+
+
+def test_flag_value_enum_no_type_error():
+    """Option with enum flag_value should not raise TypeError."""
+    try:
+        import click
+    except ImportError:
+        pytest.skip("click not available")
+
+    # This should not raise TypeError
+    @click.command()
+    @click.option("--red", "color", flag_value=Color.RED)
+    @click.option("--green", "color", flag_value=Color.GREEN)
+    def cmd(color):
+        if color is not None:
+            click.echo(color.name)
+
+    from click.testing import CliRunner
+    runner = CliRunner()
+
+    result = runner.invoke(cmd, ["--red"])
+    assert result.exit_code == 0, f"exit_code={result.exit_code}, output={result.output}"
+    assert "RED" in result.output
+
+    result = runner.invoke(cmd, ["--green"])
+    assert result.exit_code == 0
+    assert "GREEN" in result.output
+
