+def _apply_pr3363_fix():
+    """
+    Fix for pr3363: Option with non-basic flag_value should use UNPROCESSED type.

+    When flag_value is set to a non-basic type (e.g. an enum), the option's
+    type should be automatically set to UNPROCESSED instead of raising a
+    TypeError or mishandling the value during parsing.
+    """
+    try:
+        import click
+        from click.types import UNPROCESSED, STRING

+        _orig_option_init = click.Option.__init__

+        def _fixed_option_init(self, param_decls=None, show_default=False,
+                               is_flag=None, flag_value=None, multiple=False,
+                               is_eager=False, expose_value=True,
+                               is_greedy=False, help=None, hidden=False,
+                               show_choices=True, show_envvar=False, **attrs):
+            _orig_option_init(
+                self,
+                param_decls=param_decls,
+                show_default=show_default,
+                is_flag=is_flag,
+                flag_value=flag_value,
+                multiple=multiple,
+                is_eager=is_eager,
+                expose_value=expose_value,
+                is_greedy=is_greedy,
+                help=help,
+                hidden=hidden,
+                show_choices=show_choices,
+                show_envvar=show_envvar,
+                **attrs,
+            )
+            # Fix: if flag_value is non-basic and type is STRING, use UNPROCESSED
+            if (flag_value is not None
+                    and not isinstance(flag_value, (bool, int, float, str))
+                    and self.type is STRING):
+                self.type = UNPROCESSED

+        click.Option.__init__ = _fixed_option_init
+    except Exception:
+        pass
