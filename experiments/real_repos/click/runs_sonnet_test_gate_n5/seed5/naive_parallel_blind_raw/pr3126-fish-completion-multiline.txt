+ // sanitizeForFish takes a help string and returns a version safe for use
+ // in fish shell completion scripts. Fish completion uses newlines as
+ // record separators, so any newlines (and surrounding whitespace) in
+ // help text must be collapsed into a single space.
+ func sanitizeForFish(help string) string {
+ 	// Split on newlines, trim each line, and rejoin with a space.
+ 	lines := strings.Split(help, "\n")
+ 	parts := make([]string, 0, len(lines))
+ 	for _, line := range lines {
+ 		trimmed := strings.TrimSpace(line)
+ 		if trimmed != "" {
+ 			parts = append(parts, trimmed)
+ 		}
+ 	}
+ 	result := strings.Join(parts, " ")
+ 	// Also escape single quotes for fish.
+ 	result = strings.ReplaceAll(result, "'", "\\'")
+ 	return result
+ }
+ 
+ // fishCompletionEntry represents a single fish completion line.
+ type fishCompletionEntry struct {
+ 	Command     string
+ 	Short       string
+ 	Long        string
+ 	Description string
+ 	IsFlag      bool
+ 	NoFiles     bool
+ }
+ 
+ // String renders the fishCompletionEntry as a fish complete command.
+ func (e fishCompletionEntry) String() string {
+ 	var sb strings.Builder
+ 	sb.WriteString("complete")
+ 	if e.NoFiles {
+ 		sb.WriteString(" -f")
+ 	}
+ 	if e.Command != "" {
+ 		sb.WriteString(" -c ")
+ 		sb.WriteString(e.Command)
+ 	}
+ 	if e.Short != "" {
+ 		sb.WriteString(" -s ")
+ 		sb.WriteString(e.Short)
+ 	}
+ 	if e.Long != "" {
+ 		sb.WriteString(" -l ")
+ 		sb.WriteString(e.Long)
+ 	}
+ 	if e.Description != "" {
+ 		sb.WriteString(" -d '")
+ 		sb.WriteString(sanitizeForFish(e.Description))
+ 		sb.WriteString("'")
+ 	}
+ 	return sb.String()
+ }

+ // GenerateFishCompletion generates a fish completion script for the given
+ // command, its subcommands, and its flags. Help strings that span multiple
+ // lines are collapsed to a single line so that the generated script is valid.
+ func GenerateFishCompletion(commandName string, subcommands []SubCommand, flags []Flag) string {
+ 	var sb strings.Builder
+ 
+ 	// Disable file completions for the top-level command by default.
+ 	sb.WriteString(fmt.Sprintf("complete -f -c %s\n", commandName))
+ 
+ 	for _, sub := range subcommands {
+ 		entry := fishCompletionEntry{
+ 			Command:     commandName,
+ 			Long:        sub.Name,
+ 			Description: sub.Help,
+ 			NoFiles:     true,
+ 		}
+ 		sb.WriteString(entry.String())
+ 		sb.WriteString("\n")
+ 	}
+ 
+ 	for _, flag := range flags {
+ 		entry := fishCompletionEntry{
+ 			Command:     commandName,
+ 			Short:       flag.Short,
+ 			Long:        flag.Long,
+ 			Description: flag.Help,
+ 			NoFiles:     true,
+ 		}
+ 		sb.WriteString(entry.String())
+ 		sb.WriteString("\n")
+ 	}
+ 
+ 	return sb.String()
+ }
+ 
+ // SubCommand holds the name and help text for a subcommand.

+ 	Name string
+ 	Help string
+ }
+

+ type Flag struct {

+ 	Help  string
