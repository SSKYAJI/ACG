+ // sanitizeForFish takes a help string and returns a version safe for use
+ // in fish shell completion scripts. Fish completion uses newlines as
+ // record separators, so any newlines (and surrounding whitespace) in
+ // help text must be collapsed into a single space.
+ func sanitizeForFish(help string) string {
+ 	// Replace all newline characters (and optional surrounding whitespace)
+ 	// with a single space, then trim leading/trailing whitespace.
+ 	lines := strings.Split(help, "\n")
+ 	parts := make([]string, 0, len(lines))
+ 	for _, line := range lines {
+ 		trimmed := strings.TrimSpace(line)
+ 		if trimmed != "" {
+ 			parts = append(parts, trimmed)
+ 		}
+ 	}
+ 	result := strings.Join(parts, " ")
+ 	// Also escape single quotes for fish
+ 	result = strings.ReplaceAll(result, "'", "\\'")
+ 	return result
+ }
+ 
+ // fishCompletion generates a fish shell completion entry for a command or option.
+ // name is the command/option name, help is the (possibly multiline) description.
+ func fishCompletion(commandName, name, help string) string {
+ 	safeHelp := sanitizeForFish(help)
+ 	return fmt.Sprintf("complete -c %s -n '__fish_use_subcommand' -a '%s' -d '%s'",
+ 		commandName, name, safeHelp)
+ }
