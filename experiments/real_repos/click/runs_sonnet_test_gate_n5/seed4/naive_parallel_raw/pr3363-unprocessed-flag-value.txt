 def test_flag_value_non_basic_type():
     """Test that flag_value with a non-basic type (e.g. enum) sets type to UNPROCESSED."""
     import enum

     class Color(enum.Enum):
         RED = 1
         GREEN = 2

     @click.command()
     @click.option("--color", "color", flag_value=Color.RED, default=Color.GREEN)
     def cmd(color):
         click.echo(color.name)

     result = runner.invoke(cmd, ["--color"])
     assert result.exit_code == 0, result.output
     assert result.output == "RED\n"

     result = runner.invoke(cmd, [])
     assert result.exit_code == 0, result.output
     assert result.output == "GREEN\n"
