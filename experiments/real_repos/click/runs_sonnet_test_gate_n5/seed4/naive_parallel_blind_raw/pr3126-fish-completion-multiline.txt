+ // sanitizeForFish takes a help string and returns a version safe for use
+ // in fish shell completion scripts. Fish completion uses newlines as
+ // record separators, so any embedded newlines in help text must be
+ // removed or replaced to avoid generating malformed completion scripts.
+ func sanitizeForFish(help string) string {
+ 	// Replace newlines (and carriage returns) with a single space so that
+ 	// multiline help text becomes a single, readable line.
+ 	replacer := strings.NewReplacer(
+ 		"\r\n", " ",
+ 		"\r", " ",
+ 		"\n", " ",
+ 	)
+ 	sanitized := replacer.Replace(help)
+ 
+ 	// Collapse runs of whitespace that may result from the replacement.
+ 	parts := strings.Fields(sanitized)
+ 	return strings.Join(parts, " ")
+ }
+ 
+ // fishCompletionEntry represents a single fish `complete` invocation.
+ type fishCompletionEntry struct {
+ 	Command     string
+ 	ShortFlag   string
+ 	LongFlag    string
+ 	Description string
+ 	NoFiles     bool
+ }
+ 
+ // String renders the entry as a fish `complete` command.
+ func (e fishCompletionEntry) String() string {
+ 	var b strings.Builder
+ 
+ 	b.WriteString("complete")
+ 
+ 	if e.NoFiles {
+ 		b.WriteString(" -f")
+ 	}
+ 
+ 	if e.Command != "" {
+ 		fmt.Fprintf(&b, " -c %s", e.Command)
+ 	}
+ 
+ 	if e.ShortFlag != "" {
+ 		fmt.Fprintf(&b, " -s %s", e.ShortFlag)
+ 	}
+ 
+ 	if e.LongFlag != "" {
+ 		fmt.Fprintf(&b, " -l %s", e.LongFlag)
+ 	}
+ 
+ 	if e.Description != "" {
+ 		// sanitizeForFish ensures no newlines slip into the -d argument.
+ 		safe := sanitizeForFish(e.Description)
+ 		// Escape single quotes inside the description.
+ 		safe = strings.ReplaceAll(safe, "'", "\\'")
+ 		fmt.Fprintf(&b, " -d '%s'", safe)
+ 	}
+ 
+ 	return b.String()
+ }
