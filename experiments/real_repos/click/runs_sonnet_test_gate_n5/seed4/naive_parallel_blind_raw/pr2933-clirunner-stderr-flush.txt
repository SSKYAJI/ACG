+ class BytesIOCopy(io.RawIOBase):
+     """A raw IO object that writes to two streams simultaneously."""
+ 
+     def __init__(self, buffers):
+         self._buffers = buffers

+     def readable(self):
+         return False

+     def writable(self):
+         return True
+ 
+     def write(self, b):
+         for buf in self._buffers:
+             buf.write(b)
+         return len(b)
