+'use strict';
+
+/**
+ * Lexer - tokenizes markdown into block tokens.
+ *
+ * This implementation focuses on the setext heading / task list fix:
+ * - Setext headings are identified BEFORE list item paragraph processing
+ * - Checkbox detection only runs when inside a list item context
+ */
+
+const { checkbox } = require('./rules.js');
+
+// Setext heading underline pattern
+const setextHeadingRe = /^([^\n]+)\n {0,3}(=+|-+) *(?:\n+|$)/;
+// List item start
+const listItemRe = /^( {0,3})((?:[*+-]|\d{1,9}[.)]))( {1,4}| {0,1}\t)/;
+
+class Lexer {
+  constructor(options) {
+    this.options = options || {};
+    this.tokens = [];
+  }
+
+  /**
+   * Lex the source string into tokens.
+   * @param {string} src
+   * @returns {Array}
+   */
+  lex(src) {
+    src = src.replace(/\r\n|\r/g, '\n').replace(/\t/g, '    ');
+    return this._blockTokens(src, this.tokens, false);
+  }
+
+  _blockTokens(src, tokens, inListItem) {
+    while (src) {
+      // Setext heading — must be checked BEFORE paragraph/list processing
+      let cap = setextHeadingRe.exec(src);
+      if (cap) {
+        // The text of a setext heading is NEVER a task list item
+        tokens.push({
+          type: 'heading',
+          raw: cap[0],
+          depth: cap[2][0] === '=' ? 1 : 2,
+          text: cap[1].trim(),
+          // Explicitly mark: not a task list item
+          task: null,
+        });
+        src = src.slice(cap[0].length);
+        continue;
+      }
+
+      // List item
+      cap = listItemRe.exec(src);
+      if (cap) {
+        // parse list item body
+        const raw = cap[0];
+        const body = cap[0].slice(cap[1].length + cap[2].length + cap[3].length);
+        const itemTokens = [];
+        this._blockTokens(body, itemTokens, true);
+        tokens.push({
+          type: 'list_item',
+          raw,
+          tokens: itemTokens,
+        });
+        src = src.slice(raw.length);
+        continue;
+      }
+
+      // Paragraph
+      cap = /^([^\n]+(?:\n(?!setextHeading)[^\n]+)*)/.exec(src);
+      if (cap) {
+        const text = cap[1];
+        let task = null;
+        if (inListItem) {
+          const cbMatch = checkbox.exec(text);
+          if (cbMatch) {
+            task = { checked: cbMatch[1] !== ' ' };
+          }
+        }
+        tokens.push({
+          type: 'paragraph',
+          raw: cap[0],
+          text,
+          task,
+        });
+        src = src.slice(cap[0].length);
+        continue;
+      }
+
+      // Skip unrecognized character
+      src = src.slice(1);
+    }
+    return tokens;
+  }
+}
+
+module.exports = Lexer;

+'use strict';
+
+const Lexer = require('./Lexer.js');
+const Tokenizer = require('./Tokenizer.js');
+
+/**
+ * marked - minimal implementation focused on the setext/tasklist fix.
+ */
+function marked(src, options) {
+  const lexer = new Lexer(options);
+  const tokens = lexer.lex(src);
+  return render(tokens);
+}
+
+function render(tokens) {
+  let out = '';
+  for (const token of tokens) {
+    out += renderToken(token);
+  }
+  return out;
+}
+
+function renderToken(token) {
+  switch (token.type) {
+    case 'heading': {
+      const tag = `h${token.depth}`;
+      // setext heading text is rendered as-is, never as a checkbox
+      return `<${tag}>${escapeHtml(token.text)}</${tag}>\n`;
+    }
+    case 'list_item': {
+      return `<li>${render(token.tokens)}</li>\n`;
+    }
+    case 'paragraph': {
+      let text = escapeHtml(token.text);
+      if (token.task) {
+        const checked = token.task.checked ? ' checked=""' : '';
+        text = `<input${checked} disabled="" type="checkbox"> ${text.replace(/^\[[ xX]\] /, '')}`;
+      }
+      return `<p>${text}</p>\n`;
+    }
+    default:
+      return '';
+  }
+}
+
+function escapeHtml(str) {
+  return str
+    .replace(/&/g, '&amp;')
+    .replace(/</g, '&lt;')
+    .replace(/>/g, '&gt;')
+    .replace(/"/g, '&quot;');
+}
+
+marked.Lexer = Lexer;
+marked.Tokenizer = Tokenizer;
+
+module.exports = marked;
