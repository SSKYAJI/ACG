def test_templates_autoescape_enabled(tmpdir, test_client_factory):
    """Test that autoescape is enabled by default for HTML templates."""
    path = os.path.join(tmpdir, "index.html")
    with open(path, "w") as f:
        f.write("<html>{{ dangerous }}</html>")

    async def homepage(request):
        return templates.TemplateResponse(
            "index.html", {"request": request, "dangerous": "<script>alert('xss')</script>"}
        )

    app = Starlette(
        debug=True,
        routes=[Route("/", endpoint=homepage)],
    )
    templates = Jinja2Templates(directory=str(tmpdir))

    client = test_client_factory(app)
    response = client.get("/")
    # The dangerous content should be escaped
    assert "<script>" not in response.text
    assert "&lt;script&gt;" in response.text

def test_templates_with_custom_env_no_autoescape(tmpdir, test_client_factory):
    """Test that caller-supplied environments are preserved as-is."""
    path = os.path.join(tmpdir, "index.html")
    with open(path, "w") as f:
        f.write("<html>{{ dangerous }}</html>")

    async def homepage(request):
        return templates.TemplateResponse(
            "index.html", {"request": request, "dangerous": "<script>alert('xss')</script>"}
        )

    app = Starlette(
        debug=True,
        routes=[Route("/", endpoint=homepage)],
    )

    loader = jinja2.FileSystemLoader(str(tmpdir))
    # Explicitly disable autoescape in caller-supplied env
    env = jinja2.Environment(loader=loader, autoescape=False)
    templates = Jinja2Templates(env=env)

    client = test_client_factory(app)
    response = client.get("/")
    # With autoescape=False, the raw HTML should appear unescaped
    assert "<script>" in response.text
