import pytest

from starlette.applications import Starlette
from starlette.middleware.cors import CORSMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse, PlainTextResponse
from starlette.routing import Route
from starlette.testclient import TestClient


def homepage(request: Request):
    return PlainTextResponse("Homepage", status_code=200)


def test_cors_allow_all_origins(test_client_factory):
    app = Starlette(routes=[Route("/", homepage)])
    app.add_middleware(CORSMiddleware, allow_origins=["*"])
    client = test_client_factory(app)

    response = client.get("/", headers={"origin": "https://example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "*"

def test_cors_credentials_with_wildcard_origin(test_client_factory):
    """
    When allow_origins=['*'] and allow_credentials=True,
    the response should echo the request Origin instead of '*'.
    """
    app = Starlette(routes=[Route("/", homepage)])
    app.add_middleware(
        CORSMiddleware,
        allow_origins=["*"],
        allow_credentials=True,
    )
    client = test_client_factory(app)

    response = client.get("/", headers={"origin": "https://example.com"})
    assert response.status_code == 200
    # Should echo the specific origin, not '*'
    assert response.headers["access-control-allow-origin"] == "https://example.com"
    assert response.headers["access-control-allow-credentials"] == "true"

def test_cors_credentials_echoes_origin_not_wildcard(test_client_factory):
    """
    Regression test: with allow_origins=['*'] and allow_credentials=True,
    simple responses must echo the request Origin (not '*') to satisfy
    credentialed CORS requirements.
    """
    app = Starlette(routes=[Route("/", homepage)])
    app.add_middleware(
        CORSMiddleware,
        allow_origins=["*"],
        allow_credentials=True,
        allow_methods=["GET", "POST", "OPTIONS"],
    )
    client = test_client_factory(app)

    # Test with different origins
    for origin in ["https://foo.com", "https://bar.org", "http://localhost:3000"]:
        response = client.get("/", headers={"origin": origin})
        assert response.headers["access-control-allow-origin"] == origin, (
            f"Expected echoed origin for {origin}"
        )
        assert response.headers["access-control-allow-credentials"] == "true"
        # Wildcard must NOT be returned when credentials are allowed
        assert response.headers["access-control-allow-origin"] != "*"
