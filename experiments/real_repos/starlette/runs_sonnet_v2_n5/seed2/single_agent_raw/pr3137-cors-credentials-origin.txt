import pytest

from starlette.applications import Starlette
from starlette.middleware import Middleware
from starlette.middleware.cors import CORSMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse, PlainTextResponse
from starlette.routing import Route
from starlette.testclient import TestClient


def homepage(request: Request):
    return PlainTextResponse("Homepage", status_code=200)


def create_app(
    allow_origins=None,
    allow_methods=("GET",),
    allow_headers=(),
    allow_credentials=False,
    allow_origin_regex=None,
    expose_headers=(),
    max_age=600,
):
    if allow_origins is None:
        allow_origins = []
    app = Starlette(
        routes=[Route("/", homepage)],
        middleware=[
            Middleware(
                CORSMiddleware,
                allow_origins=allow_origins,
                allow_methods=allow_methods,
                allow_headers=allow_headers,
                allow_credentials=allow_credentials,
                allow_origin_regex=allow_origin_regex,
                expose_headers=expose_headers,
                max_age=max_age,
            )
        ],
    )
    return app


def test_cors_allow_all_origins():
    app = create_app(allow_origins=["*"])
    client = TestClient(app)

    response = client.get("/", headers={"origin": "https://example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "*"


def test_cors_allow_specific_origin():
    app = create_app(allow_origins=["https://example.com"])
    client = TestClient(app)

    response = client.get("/", headers={"origin": "https://example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://example.com"


def test_cors_disallow_origin():
    app = create_app(allow_origins=["https://example.com"])
    client = TestClient(app)

    response = client.get("/", headers={"origin": "https://other.com"})
    assert response.status_code == 200
    assert "access-control-allow-origin" not in response.headers


def test_cors_credentials_with_wildcard_origin():
    """
    When allow_origins=['*'] and allow_credentials=True, the response should
    echo the explicit request Origin instead of '*'.
    """
    app = create_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app)

    response = client.get("/", headers={"origin": "https://example.com"})
    assert response.status_code == 200
    # Should echo the explicit origin, not '*'
    assert response.headers["access-control-allow-origin"] == "https://example.com"
    assert response.headers["access-control-allow-credentials"] == "true"


def test_cors_credentials_with_wildcard_origin_different_origins():
    """
    When allow_origins=['*'] and allow_credentials=True, different origins
    should each get their own origin echoed back.
    """
    app = create_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app)

    response1 = client.get("/", headers={"origin": "https://example.com"})
    assert response1.headers["access-control-allow-origin"] == "https://example.com"

    response2 = client.get("/", headers={"origin": "https://other.com"})
    assert response2.headers["access-control-allow-origin"] == "https://other.com"


def test_cors_preflight_allow_all_origins():
    app = create_app(allow_origins=["*"], allow_methods=["GET", "POST"])
    client = TestClient(app)

    response = client.options(
        "/",
        headers={
            "origin": "https://example.com",
            "access-control-request-method": "GET",
        },
    )
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "*"

def test_cors_no_origin_header():

    response = client.get("/")
    assert response.status_code == 200
    assert "access-control-allow-origin" not in response.headers


def test_cors_allow_origin_regex():
    app = create_app(allow_origin_regex=r"https://.*\.example\.com")
    client = TestClient(app)

    response = client.get("/", headers={"origin": "https://sub.example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://sub.example.com"

    response = client.get("/", headers={"origin": "https://other.com"})
    assert response.status_code == 200
    assert "access-control-allow-origin" not in response.headers

def test_cors_vary_header_with_credentials():
    """
    When echoing back a specific origin (due to credentials), Vary: Origin
    should be set.
    """
    app = create_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app)

    response = client.get("/", headers={"origin": "https://example.com"})
