import pytest

from starlette.applications import Starlette
from starlette.middleware import Middleware
from starlette.middleware.sessions import SessionMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse
from starlette.routing import Route
from starlette.testclient import TestClient


def view_session(request: Request):
    return JSONResponse({"session": dict(request.session)})


def update_session(request: Request):
    request.session.update({"username": "tomchristie"})
    return JSONResponse({"session": dict(request.session)})


def clear_session(request: Request):
    request.session.clear()
    return JSONResponse({"session": dict(request.session)})


def create_app():
    app = Starlette(
        routes=[
            Route("/view", view_session),
            Route("/update", update_session),
            Route("/clear", clear_session),
        ],
        middleware=[Middleware(SessionMiddleware, secret_key="example")],
    )
    return app


def test_session():
    app = create_app()
    client = TestClient(app, raise_server_exceptions=True)

    response = client.get("/view")
    assert response.json() == {"session": {}}

    response = client.get("/update")
    assert response.json() == {"session": {"username": "tomchristie"}}

    response = client.get("/view")
    assert response.json() == {"session": {"username": "tomchristie"}}

    response = client.get("/clear")
    assert response.json() == {"session": {}}

    response = client.get("/view")
    assert response.json() == {"session": {}}


def test_session_no_cookie_for_empty_unchanged_session():
    """
    When a session is empty and unchanged, no Set-Cookie header should be sent.
    """
    app = create_app()
    client = TestClient(app, raise_server_exceptions=True)

    response = client.get("/view")
    assert response.status_code == 200
    assert response.json() == {"session": {}}
    # No Set-Cookie should be sent for an empty, unchanged session
    assert "set-cookie" not in response.headers


def test_session_cookie_set_when_modified():
    """
    When a session is modified, a Set-Cookie header should be sent.
    """
    app = create_app()
    client = TestClient(app, raise_server_exceptions=True)

    response = client.get("/update")
    assert response.status_code == 200
    assert "set-cookie" in response.headers


def test_session_no_cookie_for_read_only_access():
    """
    When a session exists but is only read (not modified), no Set-Cookie should be sent.
    """
    app = create_app()
    client = TestClient(app, raise_server_exceptions=True)

    # First, set a session
    response = client.get("/update")
    assert response.status_code == 200
    assert "set-cookie" in response.headers

    # Now read the session without modifying it
    response = client.get("/view")
    assert response.status_code == 200
    assert response.json() == {"session": {"username": "tomchristie"}}
    # No Set-Cookie should be sent for a read-only session access
    assert "set-cookie" not in response.headers


def test_session_cookie_cleared_when_session_cleared():
    """
    When a session is cleared, a Set-Cookie header should be sent to delete the cookie.
    """
    app = create_app()
    client = TestClient(app, raise_server_exceptions=True)

    # First, set a session
    client.get("/update")

    # Now clear the session
    response = client.get("/clear")
    assert response.status_code == 200
    assert response.json() == {"session": {}}
    # Set-Cookie should be sent to clear the cookie
    assert "set-cookie" in response.headers
    assert "max-age=0" in response.headers["set-cookie"].lower()


def test_session_no_cookie_when_clearing_empty_session():
    """
    When clearing an already-empty session, no Set-Cookie should be sent.
    """
    app = create_app()
    client = TestClient(app, raise_server_exceptions=True)

    # Clear a session that was never set
    response = client.get("/clear")
    assert response.status_code == 200
    assert response.json() == {"session": {}}
    # No Set-Cookie should be sent since session was already empty
    assert "set-cookie" not in response.headers


def test_session_https_only():
    app = Starlette(
        routes=[
            Route("/update", update_session),
            Route("/view", view_session),
        ],
        middleware=[
            Middleware(SessionMiddleware, secret_key="example", https_only=True)
        ],
    )
    client = TestClient(app, base_url="https://testserver", raise_server_exceptions=True)

    response = client.get("/update")
    assert response.status_code == 200
    assert "secure" in response.headers.get("set-cookie", "")


def test_session_same_site_strict():
    app = Starlette(
        routes=[
            Route("/update", update_session),
        ],
        middleware=[
            Middleware(SessionMiddleware, secret_key="example", same_site="strict")
        ],
    )
    client = TestClient(app, raise_server_exceptions=True)

    response = client.get("/update")
    assert response.status_code == 200
    assert "samesite=strict" in response.headers.get("set-cookie", "")


def test_concurrent_requests_read_only():
    """
    Regression test: concurrent read-only requests should not send Set-Cookie.
    """
    app = create_app()
    client = TestClient(app, raise_server_exceptions=True)

    # Set up a session first
    client.get("/update")

    # Multiple read-only requests should not set cookies
    for _ in range(3):
        response = client.get("/view")
        assert response.status_code == 200
        assert "set-cookie" not in response.headers
