import pytest

from starlette.applications import Starlette
from starlette.middleware.cors import CORSMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse
from starlette.routing import Route
from starlette.testclient import TestClient


def homepage(request: Request):
    return JSONResponse({"hello": "world"})


# ---------------------------------------------------------------------------
# Helper to build a simple app with CORS middleware
# ---------------------------------------------------------------------------

def make_app(**cors_kwargs):
    app = Starlette(routes=[Route("/", homepage)])
    app.add_middleware(CORSMiddleware, **cors_kwargs)
    return app


# ---------------------------------------------------------------------------
# Tests
# ---------------------------------------------------------------------------

def test_cors_allow_all_origins_simple():
    """With allow_origins=['*'] and no credentials, response should have '*'."""
    app = make_app(allow_origins=["*"])
    client = TestClient(app, raise_server_exceptions=True)
    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] in ("*", "https://example.com")


def test_cors_credentials_echoes_origin():
    """
    With allow_origins=['*'] and allow_credentials=True, the simple response
    must echo the explicit request Origin instead of '*'.
    """
    app = make_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app, raise_server_exceptions=True)
    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    # Must NOT be wildcard when credentials are allowed
    assert response.headers["access-control-allow-origin"] == "https://example.com"
    assert response.headers.get("access-control-allow-credentials") == "true"


def test_cors_credentials_preflight_echoes_origin():
    """
    Preflight with allow_origins=['*'] and allow_credentials=True should also
    echo the explicit Origin.
    """
    app = make_app(
        allow_origins=["*"],
        allow_credentials=True,
        allow_methods=["GET", "POST"],
    )
    client = TestClient(app, raise_server_exceptions=True)
    response = client.options(
        "/",
        headers={
            "Origin": "https://example.com",
            "Access-Control-Request-Method": "GET",
        },
    )
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://example.com"
    assert response.headers.get("access-control-allow-credentials") == "true"


def test_cors_specific_origin():
    """Specific origin in allow_origins should be echoed back."""
    app = make_app(allow_origins=["https://example.com"])
    client = TestClient(app, raise_server_exceptions=True)
    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://example.com"


def test_cors_disallowed_origin():
    """Requests from disallowed origins should not get ACAO header."""
    app = make_app(allow_origins=["https://allowed.com"])
    client = TestClient(app, raise_server_exceptions=True)
    response = client.get("/", headers={"Origin": "https://evil.com"})
    assert response.status_code == 200
    # Should not grant access
    acao = response.headers.get("access-control-allow-origin", "")
    assert acao != "https://evil.com"


def test_cors_no_origin_header():
    """Requests without Origin header should pass through unchanged."""
    app = make_app(allow_origins=["*"])
    client = TestClient(app, raise_server_exceptions=True)
    response = client.get("/")
    assert response.status_code == 200
    assert "access-control-allow-origin" not in response.headers
