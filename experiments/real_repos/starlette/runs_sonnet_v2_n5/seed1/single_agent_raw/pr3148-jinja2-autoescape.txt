@pytest.mark.skipif(jinja2 is None, reason="jinja2 not installed")
def test_templates_with_directory(template_dir):
    (template_dir / "index.html").write_text("<html>Hello {{ name }}</html>")

    templates = Jinja2Templates(directory=str(template_dir))

    def homepage(request):
        return templates.TemplateResponse("index.html", {"request": request, "name": "World"})

    app = Starlette(routes=[Route("/", homepage)])
    client = TestClient(app)
    response = client.get("/")
    assert response.status_code == 200
    assert "Hello World" in response.text


@pytest.mark.skipif(jinja2 is None, reason="jinja2 not installed")
def test_templates_autoescape_enabled(template_dir):
    """Test that HTML templates have autoescape enabled by default."""
    (template_dir / "index.html").write_text("<html>{{ name }}</html>")

    templates = Jinja2Templates(directory=str(template_dir))

    def homepage(request):
        return templates.TemplateResponse(
            "index.html", {"request": request, "name": "<script>alert('xss')</script>"}
        )

    app = Starlette(routes=[Route("/", homepage)])
    client = TestClient(app)
    response = client.get("/")
    assert response.status_code == 200
    # The script tag should be escaped
    assert "<script>" not in response.text
    assert "&lt;script&gt;" in response.text

@pytest.mark.skipif(jinja2 is None, reason="jinja2 not installed")
def test_templates_with_custom_env(template_dir):
    """Test that a caller-supplied environment is preserved as-is."""
    (template_dir / "index.html").write_text("<html>{{ name }}</html>")

    # Create env WITHOUT autoescape
    loader = jinja2.FileSystemLoader(str(template_dir))
    env = jinja2.Environment(loader=loader, autoescape=False)

    templates = Jinja2Templates(env=env)

    def homepage(request):
        return templates.TemplateResponse(
            "index.html",
            {"request": request, "name": "<b>raw</b>"},
        )

    app = Starlette(routes=[Route("/", homepage)])
    client = TestClient(app)
    response = client.get("/")
    assert response.status_code == 200
    # Custom env has autoescape=False, so raw HTML should pass through
    assert "<b>raw</b>" in response.text
