import pytest

from starlette.applications import Starlette
from starlette.middleware.sessions import SessionMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse
from starlette.routing import Route
from starlette.testclient import TestClient


def view_session(request: Request):
    return JSONResponse({"session": dict(request.session)})


def update_session(request: Request):
    request.session.update({"user": "alice"})
    return JSONResponse({"session": dict(request.session)})


def clear_session(request: Request):
    request.session.clear()
    return JSONResponse({"session": dict(request.session)})


def read_session(request: Request):
    # Only reads, does not mutate
    _ = request.session.get("user")
    return JSONResponse({"session": dict(request.session)})


app = Starlette(
    routes=[
        Route("/view", view_session),
        Route("/update", update_session),
        Route("/clear", clear_session),
        Route("/read", read_session),
    ]
)
app.add_middleware(SessionMiddleware, secret_key="test-secret")


def test_session_set():
    client = TestClient(app, raise_server_exceptions=True)
    response = client.get("/update")
    assert response.status_code == 200
    assert "session" in response.cookies
    assert response.json() == {"session": {"user": "alice"}}


def test_session_read_only_does_not_set_cookie():
    """Reading an existing session without mutation should not send Set-Cookie."""
    client = TestClient(app, raise_server_exceptions=True)
    # First, establish a session
    response = client.get("/update")
    assert "session" in response.cookies

    # Now read without mutating – should NOT get a new Set-Cookie header
    response2 = client.get("/read")
    assert response2.status_code == 200
    assert "set-cookie" not in response2.headers


def test_session_clear():
    """Clearing a session should send a cookie deletion header."""
    client = TestClient(app, raise_server_exceptions=True)
    # Establish session
    client.get("/update")
    # Clear it
    response = client.get("/clear")
    assert response.status_code == 200
    assert "set-cookie" in response.headers
    assert "Max-Age=0" in response.headers["set-cookie"]


def test_empty_session_no_cookie():
    """Accessing an empty session without mutation should not set a cookie."""
    client = TestClient(app, raise_server_exceptions=True)
    response = client.get("/view")
    assert response.status_code == 200
    assert "set-cookie" not in response.headers


def test_session_update_sends_cookie():
    """Mutating a session should always send Set-Cookie."""
    client = TestClient(app, raise_server_exceptions=True)
    response = client.get("/update")
    assert response.status_code == 200
    assert "set-cookie" in response.headers


def test_concurrent_read_only_sessions():
    """
    Simulate two concurrent requests that both read the same session
    without mutating it. Neither should send Set-Cookie.
    """
    client = TestClient(app, raise_server_exceptions=True)
    # Establish session
    client.get("/update")

    # Two read-only requests
    r1 = client.get("/read")
    r2 = client.get("/read")
    assert "set-cookie" not in r1.headers
    assert "set-cookie" not in r2.headers


def test_session_cookie_subpath():
    """Cookie path should always be '/'."""
    client = TestClient(app, raise_server_exceptions=True)
    response = client.get("/update")
    assert "path=/" in response.headers.get("set-cookie", "")
