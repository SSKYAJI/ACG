+ // encode serializes and signs the session into a cookie value string.
+ func encode(s *Session, secret []byte) (string, error) {
+ 	payload := cookiePayload{
+ 		ID:   s.ID(),
+ 		Data: s.snapshot(),
+ 	}
+ 	raw, err := json.Marshal(payload)
+ 	if err != nil {
+ 		return "", fmt.Errorf("session: encode marshal: %w", err)
+ 	}
+ 	encoded := base64.RawURLEncoding.EncodeToString(raw)
+ 	sig := sign(encoded, secret)
+ 	return encoded + "." + sig, nil
+ }

+ // decode verifies the signature and deserializes the cookie value.
+ func decode(value string, secret []byte) (*cookiePayload, error) {
+ 	parts := strings.SplitN(value, ".", 2)
+ 	if len(parts) != 2 {
+ 		return nil, fmt.Errorf("session: decode: invalid format")
+ 	}
+ 	encoded, sig := parts[0], parts[1]
+ 	expected := sign(encoded, secret)
+ 	if !hmac.Equal([]byte(sig), []byte(expected)) {
+ 		return nil, fmt.Errorf("session: decode: invalid signature")
+ 	}
+ 	raw, err := base64.RawURLEncoding.DecodeString(encoded)
+ 	if err != nil {
+ 		return nil, fmt.Errorf("session: decode: base64: %w", err)
+ 	}
+ 	var payload cookiePayload
+ 	if err := json.Unmarshal(raw, &payload); err != nil {
+ 		return nil, fmt.Errorf("session: decode: unmarshal: %w", err)
+ 	}
+ 	return &payload, nil
+ }

+ // sign computes an HMAC-SHA256 signature for the given data.
+ func sign(data string, secret []byte) string {
+ 	mac := hmac.New(sha256.New, secret)
+ 	mac.Write([]byte(data))
+ 	return base64.RawURLEncoding.EncodeToString(mac.Sum(nil))
+ }

+ // loadOrCreate loads a session from the request cookie, or creates a new one.
+ func (m *Middleware) loadOrCreate(r *http.Request) (*Session, error) {
+ 	cookie, err := r.Cookie(m.opts.CookieName)
+ 	if err != nil {
+ 		// No cookie present; create new session.
+ 		return newSession()
+ 	}
+ 	payload, err := decode(cookie.Value, m.opts.Secret)
+ 	if err != nil {
+ 		// Invalid/tampered cookie; create new session.
+ 		return newSession()
+ 	}
+ 	return loadSession(payload.ID, payload.Data), nil
+ }

+ // saveCookie writes the Set-Cookie header if necessary.
+ //
+ // Rules:
+ //  1. If the session was cleared → send an expiring cookie to delete it.
+ //  2. If the session was mutated (but not cleared) → send updated cookie.
+ //  3. If the session is new and was never mutated → do NOT send a cookie
+ //     (avoid creating empty sessions for every request).
+ //  4. If the session is not new and was not mutated → do NOT send a cookie
+ //     (avoid refreshing unchanged sessions).
+ func (m *Middleware) saveCookie(w http.ResponseWriter, s *Session) {
+ 	if s.IsCleared() {
+ 		// Delete the cookie by setting MaxAge=-1 and an expired time.
+ 		http.SetCookie(w, &http.Cookie{
+ 			Name:     m.opts.CookieName,
+ 			Value:    "",
+ 			Path:     m.opts.Path,
+ 			Domain:   m.opts.Domain,
+ 			MaxAge:   -1,
+ 			Expires:  time.Unix(0, 0),
+ 			HttpOnly: m.opts.HttpOnly,
+ 			Secure:   m.opts.Secure,
+ 			SameSite: m.opts.SameSite,
+ 		})
+ 		return
+ 	}
+
+ 	if !s.IsMutated() {
+ 		// No changes; skip Set-Cookie entirely.
+ 		return
+ 	}
+
+ 	// Session was mutated; encode and send.
+ 	value, err := encode(s, m.opts.Secret)
+ 	if err != nil {
+ 		// Encoding failed; don't send a broken cookie.
+ 		return
+ 	}
+
+ 	cookie := &http.Cookie{
+ 		Name:     m.opts.CookieName,
+ 		Value:    value,
+ 		Path:     m.opts.Path,
+ 		Domain:   m.opts.Domain,
+ 		MaxAge:   m.opts.MaxAge,
+ 		HttpOnly: m.opts.HttpOnly,
+ 		Secure:   m.opts.Secure,
+ 		SameSite: m.opts.SameSite,
+ 	}
+ 	if m.opts.MaxAge > 0 {
+ 		cookie.Expires = time.Now().Add(time.Duration(m.opts.MaxAge) * time.Second)
+ 	}
+ 	http.SetCookie(w, cookie)
+ }

+ // responseWriter buffers the status code and headers so we can inject
+ // the Set-Cookie header before the response is actually written.
+ type responseWriter struct {
+ 	http.ResponseWriter
+ 	code        int
+ 	wroteHeader bool
+ 	body        []byte
+ }
+
+ func (rw *responseWriter) WriteHeader(code int) {
+ 	if !rw.wroteHeader {
+ 		rw.code = code
+ 		rw.wroteHeader = true
+ 	}
+ }
+
+ func (rw *responseWriter) Write(b []byte) (int, error) {
+ 	rw.body = append(rw.body, b...)
+ 	return len(b), nil
+ }

+ // flush sends the buffered response to the underlying ResponseWriter.
+ func (rw *responseWriter) flush() {
+ 	rw.ResponseWriter.WriteHeader(rw.code)
+ 	if len(rw.body) > 0 {
+ 		rw.ResponseWriter.Write(rw.body) //nolint:errcheck
+ 	}
