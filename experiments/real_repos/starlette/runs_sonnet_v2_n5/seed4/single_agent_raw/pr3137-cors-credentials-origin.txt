import sys
import os
import pytest

sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))


async def simple_app(scope, receive, send):
    body = b"Hello"
    await send({
        "type": "http.response.start",
        "status": 200,
        "headers": [
            [b"content-type", b"text/plain"],
            [b"content-length", str(len(body)).encode()],
        ],
    })
    await send({"type": "http.response.body", "body": body})


def make_scope(method="GET", path="/", headers=None):
    return {
        "type": "http",
        "method": method,
        "path": path,
        "headers": headers or [],
    }


async def run_request(app, scope):
    messages = []

    async def receive():
        return {"type": "http.request", "body": b"", "more_body": False}

    async def send(message):
        messages.append(message)

    await app(scope, receive, send)
    return messages


def get_response_headers(messages):
    for msg in messages:
        if msg.get("type") == "http.response.start":
            return {
                k.decode("latin-1").lower(): v.decode("latin-1")
                for k, v in msg.get("headers", [])
            }
    return {}


@pytest.mark.asyncio
async def test_allow_all_origins_no_credentials():
    """With allow_origins=['*'] and no credentials, should return '*'."""
    from starlette.middleware.cors import CORSMiddleware

    app = CORSMiddleware(simple_app, allow_origins=["*"])
    scope = make_scope(headers=[[b"origin", b"https://example.com"]])
    messages = await run_request(app, scope)
    headers = get_response_headers(messages)
    assert headers.get("access-control-allow-origin") == "*"


@pytest.mark.asyncio
async def test_allow_all_origins_with_credentials_echoes_origin():
    """With allow_origins=['*'] and allow_credentials=True, should echo the request Origin."""
    from starlette.middleware.cors import CORSMiddleware

    app = CORSMiddleware(simple_app, allow_origins=["*"], allow_credentials=True)
    scope = make_scope(headers=[[b"origin", b"https://example.com"]])
    messages = await run_request(app, scope)
    headers = get_response_headers(messages)
    # Should echo the specific origin, not '*'
    assert headers.get("access-control-allow-origin") == "https://example.com"
    assert headers.get("access-control-allow-credentials") == "true"


@pytest.mark.asyncio
async def test_allow_all_origins_with_credentials_different_origin():
    """Different origins should each be echoed back."""
    from starlette.middleware.cors import CORSMiddleware

    app = CORSMiddleware(simple_app, allow_origins=["*"], allow_credentials=True)

    for origin in ["https://foo.com", "https://bar.org", "http://localhost:3000"]:
        scope = make_scope(headers=[[b"origin", origin.encode()]])
        messages = await run_request(app, scope)
        headers = get_response_headers(messages)
        assert headers.get("access-control-allow-origin") == origin, \
            f"Expected {origin}, got {headers.get('access-control-allow-origin')}"


@pytest.mark.asyncio
async def test_preflight_with_credentials_echoes_origin():
    """Preflight with credentials should echo the specific origin."""
    from starlette.middleware.cors import CORSMiddleware

    app = CORSMiddleware(
        simple_app,
        allow_origins=["*"],
        allow_credentials=True,
        allow_methods=["GET", "POST"],
    )
    scope = make_scope(
        method="OPTIONS",
        headers=[
            [b"origin", b"https://example.com"],
            [b"access-control-request-method", b"POST"],
        ],
    )
    messages = await run_request(app, scope)
    headers = get_response_headers(messages)
    assert headers.get("access-control-allow-origin") == "https://example.com"
    assert headers.get("access-control-allow-credentials") == "true"
