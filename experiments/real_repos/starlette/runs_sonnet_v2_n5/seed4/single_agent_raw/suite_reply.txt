class _TemplateResponse:
    def __init__(
        self,
        template: typing.Any,
        context: dict,
        status_code: int = 200,
        headers: typing.Optional[dict] = None,
        media_type: str = "text/html",
        background: typing.Any = None,
    ):
        self.template = template
        self.context = context
        self.status_code = status_code
        self.headers = headers or {}
        self.media_type = media_type
        self.background = background

    async def __call__(self, scope: dict, receive: typing.Any, send: typing.Any) -> None:
        prefix = "charset="
        charset = "utf-8"
        content = self.template.render(self.context)
        body = content.encode(charset)
        await send(
            {
                "type": "http.response.start",
                "status": self.status_code,
                "headers": [
                    [b"content-type", f"{self.media_type}; {prefix}{charset}".encode()],
                    [b"content-length", str(len(body)).encode()],
                ]
                + [
                    [k.encode(), v.encode()]
                    for k, v in self.headers.items()
                ],
            }
        )
        await send({"type": "http.response.body", "body": body})
        if self.background is not None:
            await self.background()

class _SessionDict(dict):
    """
    A dict subclass that tracks whether it has been modified.
    """

    def __init__(self, *args: typing.Any, **kwargs: typing.Any) -> None:
        super().__init__(*args, **kwargs)
        self._modified = False
        self._accessed = False

    @property
    def is_modified(self) -> bool:
        return self._modified

    @property
    def is_accessed(self) -> bool:
        return self._accessed

    def __setitem__(self, key: typing.Any, value: typing.Any) -> None:
        self._modified = True
        self._accessed = True
        super().__setitem__(key, value)

    def __delitem__(self, key: typing.Any) -> None:
        self._modified = True
        self._accessed = True
        super().__delitem__()

    def __getitem__(self, key: typing.Any) -> typing.Any:
        self._accessed = True
        return super().__getitem__(key)

    def get(self, key: typing.Any, default: typing.Any = None) -> typing.Any:
        self._accessed = True
        return super().get(key, default)

    def __contains__(self, key: typing.Any) -> bool:
        self._accessed = True
        return super().__contains__(key)

    def clear(self) -> None:
        self._modified = True
        self._accessed = True
        super().clear()

    def update(self, *args: typing.Any, **kwargs: typing.Any) -> None:  # type: ignore[override]
        self._modified = True
        self._accessed = True
        super().update(*args, **kwargs)

    def pop(self, *args: typing.Any) -> typing.Any:
        self._modified = True
        self._accessed = True
        return super().pop(*args)

    def setdefault(self, key: typing.Any, default: typing.Any = None) -> typing.Any:

        self._accessed = True

import re
import typing
from starlette.datastructures import Headers
from starlette.responses import PlainTextResponse, Response
from starlette.types import ASGIApp, Receive, Scope, Send

ALL_METHODS = ("DELETE", "GET", "HEAD", "OPTIONS", "PATCH", "POST", "PUT")
SAFELISTED_HEADERS = {"Accept", "Accept-Language", "Content-Language", "Content-Type"}


class CORSMiddleware:
    def __init__(
        self,
        app: typing.Any,
        allow_origins: typing.Sequence[str] = (),
        allow_methods: typing.Sequence[str] = ("GET",),
        allow_headers: typing.Sequence[str] = (),
        allow_credentials: bool = False,
        allow_origin_regex: typing.Optional[str] = None,
        expose_headers: typing.Sequence[str] = (),
        max_age: int = 600,
    ) -> None:
        if allow_origins and "*" in allow_origins:
            self.allow_all_origins = True
        else:
            self.allow_all_origins = False

        compiled_allow_origin_regex = None
        if allow_origin_regex is not None:
            compiled_allow_origin_regex = re.compile(allow_origin_regex)

        self.app = app
        self.allow_origins = list(allow_origins)
        self.allow_all_headers = "*" in allow_headers
        self.allow_headers = [h.lower() for h in allow_headers]
        self.allow_all_methods = "*" in allow_methods
        self.allow_methods = list(allow_methods)
        self.allow_credentials = allow_credentials
        self.allow_origin_regex = compiled_allow_origin_regex
        self.expose_headers = list(expose_headers)
        self.max_age = max_age

    async def __call__(self, scope: dict, receive: typing.Any, send: typing.Any) -> None:
        if scope["type"] != "http":
            await self.app(scope, receive, send)
            return

        headers = Headers(scope=scope)
        origin = headers.get("origin")

        if origin is None:
            await self.app(scope, receive, send)
            return

        if scope["method"] == "OPTIONS" and "access-control-request-method" in headers:
            response = self.preflight_response(request_headers=headers)
            await response(scope, receive, send)
            return

        await self.simple_response(scope, receive, send, request_headers=headers)

    def is_allowed_origin(self, origin: str) -> bool:
        if self.allow_all_origins:
            return True
        if origin in self.allow_origins:
            return True
        if self.allow_origin_regex is not None and self.allow_origin_regex.fullmatch(origin):
            return True
        return False

    def preflight_response(self, request_headers: "Headers") -> typing.Any:
        requested_origin = request_headers["origin"]
        requested_method = request_headers.get("access-control-request-method")
        requested_headers = request_headers.get("access-control-request-headers")

        headers: typing.Dict[str, str] = {}
        failures = []

        if self.is_allowed_origin(requested_origin):
            # For credentials, we must echo the specific origin
            if self.allow_credentials:
                headers["access-control-allow-origin"] = requested_origin
            elif self.allow_all_origins:
                headers["access-control-allow-origin"] = "*"
            else:
                headers["access-control-allow-origin"] = requested_origin
        else:
            failures.append("origin")

        if requested_method and (
            self.allow_all_methods or requested_method in self.allow_methods
        ):
            if self.allow_all_methods:
                headers["access-control-allow-methods"] = ", ".join(ALL_METHODS)
            else:
                headers["access-control-allow-methods"] = ", ".join(self.allow_methods)
        elif requested_method:
            failures.append("method")

        if requested_headers:
            if self.allow_all_headers:
                headers["access-control-allow-headers"] = requested_headers
            else:
                # Check each requested header
                req_headers = [h.strip().lower() for h in requested_headers.split(",")]
                for h in req_headers:
                    if h not in self.allow_headers and h not in {s.lower() for s in SAFELISTED_HEADERS}:
                        failures.append("headers")
                        break
                else:
                    headers["access-control-allow-headers"] = requested_headers

        if self.allow_credentials:
            headers["access-control-allow-credentials"] = "true"

        if self.max_age:
            headers["access-control-max-age"] = str(self.max_age)

        if failures:
            failure_text = "Disallowed CORS " + ", ".join(failures)
            return PlainTextResponse(failure_text, status_code=400, headers=headers)

        return PlainTextResponse("OK", status_code=200, headers=headers)

    async def simple_response(
        self,
        scope: dict,
        receive: typing.Any,

import typing


class Headers:
    """
    An immutable, case-insensitive multi-dict for HTTP headers.
    """

    def __init__(
        self,
        headers: typing.Optional[typing.List[typing.Tuple[bytes, bytes]]] = None,
        scope: typing.Optional[dict] = None,
    ) -> None:
        self._list: typing.List[typing.Tuple[bytes, bytes]]
        if headers is not None:
            self._list = headers
        elif scope is not None:
            self._list = scope.get("headers", [])
        else:
            self._list = []

    def __getitem__(self, key: str) -> str:
        key_lower = key.lower().encode("latin-1")
        for header_key, header_value in self._list:
            if header_key.lower() == key_lower:
                return header_value.decode("latin-1")
        raise KeyError(key)

    def get(self, key: str, default: typing.Optional[str] = None) -> typing.Optional[str]:
        try:
            return self[key]
        except KeyError:
            return default

    def __contains__(self, key: object) -> bool:
        if not isinstance(key, str):
            return False
        key_lower = key.lower().encode("latin-1")
        for header_key, _ in self._list:
            if header_key.lower() == key_lower:
                return True
        return False

    def items(self) -> typing.List[typing.Tuple[str, str]]:
        return [
            (k.decode("latin-1"), v.decode("latin-1"))
            for k, v in self._list
        ]

    def keys(self) -> typing.List[str]:
        return [k.decode("latin-1") for k, _ in self._list]

    def values(self) -> typing.List[str]:
        return [v.decode("latin-1") for _, v in self._list]

import typing
import json as json_module


class Response:
    media_type: typing.Optional[str] = None
    charset = "utf-8"

    def __init__(
        self,
        content: typing.Any = None,
        status_code: int = 200,
        headers: typing.Optional[dict] = None,
        media_type: typing.Optional[str] = None,
        background: typing.Any = None,
    ) -> None:
        self.status_code = status_code
        if media_type is not None:
            self.media_type = media_type
        self.background = background
        self.body = self.render(content)
        self.raw_headers = self.init_headers(headers)

    def render(self, content: typing.Any) -> bytes:
        if content is None:
            return b""
        if isinstance(content, bytes):
            return content
        return content.encode(self.charset)

    def init_headers(self, headers: typing.Optional[dict] = None) -> typing.List[typing.List[bytes]]:
        raw_headers: typing.List[typing.List[bytes]] = []
        populate_content_length = True
        populate_content_type = True

        if headers:
            for k, v in headers.items():
                raw_headers.append([k.lower().encode("latin-1"), v.encode("latin-1")])
                if k.lower() == "content-length":
                    populate_content_length = False
                if k.lower() == "content-type":
                    populate_content_type = False

        body = getattr(self, "body", b"")
        if populate_content_length:
            raw_headers.append([b"content-length", str(len(body)).encode("latin-1")])

        if populate_content_type and self.media_type is not None:
            content_type = self.media_type
            if content_type.startswith("text/") and "charset=" not in content_type:
                content_type += "; charset=" + self.charset
            raw_headers.append([b"content-type", content_type.encode("latin-1")])

        return raw_headers

    async def __call__(self, scope: dict, receive: typing.Any, send: typing.Any) -> None:
        await send(
            {
                "type": "http.response.start",
                "status": self.status_code,
                "headers": self.raw_headers,
            }
        )
        await send({"type": "http.response.body", "body": self.body})
        if self.background is not None:
            await self.background()
