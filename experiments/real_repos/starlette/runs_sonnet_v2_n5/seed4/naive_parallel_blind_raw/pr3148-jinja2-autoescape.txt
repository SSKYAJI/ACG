+ class _TemplateResponse:
+     def __init__(self, template, context, status_code=200, headers=None, media_type=None, background=None):
+         self.template = template
+         self.context = context
+         self.status_code = status_code
+         self.headers = headers or {}
+         self.media_type = media_type or "text/html"
+         self.background = background
+         self.body = template.render(context).encode("utf-8")
+ 
+     async def __call__(self, scope, receive, send):
+         prefix = "HTTP/"
+         await send(
+             {
+                 "type": "http.response.start",
+                 "status": self.status_code,
+                 "headers": [
+                     [b"content-type", (self.media_type + "; charset=utf-8").encode("latin-1")],
+                     [b"content-length", str(len(self.body)).encode("latin-1")],
+                 ],
+             }
+         )
+         await send({"type": "http.response.body", "body": self.body})
+         if self.background is not None:
+             await self.background()

+     def __init__(
+         self,
+         directory: typing.Union[str, None] = None,
+         *,
+         env: typing.Union[Environment, None] = None,
+     ) -> None:
+         if env is not None:
+             self.env = env
+         elif directory is not None:
+             self.env = self._create_env(directory)
+         else:
+             raise ValueError("Either 'directory' or 'env' must be provided.")
+ 
+     def _create_env(self, directory: str) -> Environment:
+         loader = FileSystemLoader(directory)
+         env = Environment(
+             loader=loader,
+             autoescape=select_autoescape(),
+         )
+         return env
+ 
+     def get_template(self, name: str):
+         return self.env.get_template(name)
+ 
+     def TemplateResponse(
+         self,

+         status_code: int = 200,
+         headers: dict = None,
+         media_type: str = None,
+         background=None,
+     ) -> _TemplateResponse:
