+ // Session holds the session data and tracks whether it has been mutated.
+ type Session struct {
+ 	mu      sync.RWMutex
+ 	data    map[string]interface{}
+ 	dirty   bool
+ 	cleared bool
+ 	id      string
+ }

+ // newSession creates a new empty session with a freshly generated ID.
+ func newSession() (*Session, error) {
+ 	id, err := generateID()
+ 	if err != nil {
+ 		return nil, err
+ 	}
+ 	return &Session{
+ 		data: make(map[string]interface{}),
+ 		id:   id,
+ 	}, nil
+ }

+ // loadSession creates a Session from previously decoded data.
+ func loadSession(id string, data map[string]interface{}) *Session {
+ 	return &Session{
+ 		data: data,
+ 		id:   id,
+ 	}
+ }
+ 
+ // ID returns the session identifier.
+ func (s *Session) ID() string {
+ 	s.mu.RLock()
+ 	defer s.mu.RUnlock()
+ 	return s.id
+ }
+ 
+ // Get retrieves a value from the session. This is a read-only operation
+ // and does NOT mark the session as dirty.
+ func (s *Session) Get(key string) (interface{}, bool) {
+ 	s.mu.RLock()
+ 	defer s.mu.RUnlock()
+ 	v, ok := s.data[key]
+ 	return v, ok
+ }
+ 
+ // Set stores a value in the session and marks the session as dirty (mutated).
+ func (s *Session) Set(key string, value interface{}) {
+ 	s.mu.Lock()
+ 	defer s.mu.Unlock()
+ 	s.data[key] = value
+ 	s.dirty = true
+ }
+ 
+ // Delete removes a key from the session and marks the session as dirty.
+ func (s *Session) Delete(key string) {
+ 	s.mu.Lock()
+ 	defer s.mu.Unlock()
+ 	delete(s.data, key)
+ 	s.dirty = true
+ }

+ // Clear removes all data from the session and marks it for cookie removal.
+ // After Clear, the middleware will send a Set-Cookie header that expires
+ // the cookie immediately, regardless of whether the session was previously

+ func (s *Session) Clear() {
+ 	s.mu.Lock()
+ 	defer s.mu.Unlock()
+ 	s.data = make(map[string]interface{})

+ 	s.cleared = true

+ // Handler wraps the provided http.Handler with session management.
+ //
+ // Behavior:
+ //   - Loads the session from the signed cookie (or creates a new one).
+ //   - Injects the *Session into the request context.
+ //   - After the inner handler returns:
+ //     - If the session was cleared, sends an expiring Set-Cookie header.
+ //     - If the session was mutated (dirty), encodes and sends a new Set-Cookie.
+ //     - If the session was only read (not dirty), no Set-Cookie is sent.
+ func (m *Middleware) Handler(next http.Handler) http.Handler {
+ 	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
+ 		sess, err := m.loadFromRequest(r)
+ 		if err != nil {
+ 			// Could not decode existing cookie; start a fresh session.
+ 			sess, err = newSession()
+ 			if err != nil {
+ 				http.Error(w, "internal server error", http.StatusInternalServerError)
+ 				return
+ 			}
+ 		}
+ 
+ 		ctx := context.WithValue(r.Context(), sessionContextKey, sess)
+ 		next.ServeHTTP(w, r.WithContext(ctx))
+ 
+ 		// Post-handler: decide whether to write a Set-Cookie header.
+ 		if sess.IsCleared() {
+ 			m.clearCookie(w)
+ 			return
+ 		}
+ 		if sess.IsDirty() {
+ 			if err := m.saveCookie(w, sess); err != nil {
+ 				// Best-effort: log would go here in production code.
+ 				_ = err
+ 			}
+ 		}
+ 		// If neither cleared nor dirty, no Set-Cookie is written.
+ 	})
+ }

+ // Returns nil if no session is present.

+ 	s, _ := ctx.Value(sessionContextKey).(*Session)

+ func (m *Middleware) loadFromRequest(r *http.Request) (*Session, error) {
+ 	cookie, err := r.Cookie(m.opts.cookieName())
+ 	if err != nil {
+ 		return nil, ErrSessionNotFound
+ 	}
+ 	payload, err := m.decode(cookie.Value)
+ 	if err != nil {
+ 		return nil, err
+ 	}
+ 	return loadSession(payload.ID, payload.Data), nil
+ }
+ 
+ func (m *Middleware) saveCookie(w http.ResponseWriter, sess *Session) error {
+ 	sess.mu.RLock()
+ 	payload := cookiePayload{
+ 		ID:   sess.id,
+ 		Data: sess.data,
+ 	}
+ 	sess.mu.RUnlock()
+ 
+ 	value, err := m.encode(payload)
+ 	if err != nil {
+ 		return fmt.Errorf("session: encode: %w", err)
+ 	}
+ 
+ 	http.SetCookie(w, &http.Cookie{
+ 		Name:     m.opts.cookieName(),
+ 		Value:    value,
+ 		Path:     m.opts.cookiePath(),
+ 		Domain:   m.opts.Domain,
+ 		MaxAge:   m.opts.MaxAge,
+ 		Secure:   m.opts.Secure,
+ 		HttpOnly: m.opts.HttpOnly,
+ 		SameSite: m.opts.SameSite,
+ 	})
+ 	return nil
+ }
+ 
+ func (m *Middleware) clearCookie(w http.ResponseWriter) {
+ 	http.SetCookie(w, &http.Cookie{
+ 		Name:     m.opts.cookieName(),
+ 		Value:    "",
+ 		Path:     m.opts.cookiePath(),
+ 		Domain:   m.opts.Domain,
+ 		MaxAge:   -1,
+ 		Expires:  time.Unix(0, 0),
+ 		Secure:   m.opts.Secure,
+ 		HttpOnly: m.opts.HttpOnly,
+ 		SameSite: m.opts.SameSite,
+ 	})
+ }
+ 
+ // encode serialises and signs the payload.
+ // Format: base64(json) + "." + base64(hmac)
+ func (m *Middleware) encode(p cookiePayload) (string, error) {
+ 	raw, err := json.Marshal(p)
+ 	if err != nil {
+ 		return "", err
+ 	}
+ 	encoded := base64.RawURLEncoding.EncodeToString(raw)
+ 	sig := m.sign(encoded)
+ 	return encoded + "." + sig, nil
+ }
+ 
+ // decode verifies the signature and deserialises the payload.
+ func (m *Middleware) decode(value string) (cookiePayload, error) {
+ 	parts := strings.SplitN(value, ".", 2)
+ 	if len(parts) != 2 {
+ 		return cookiePayload{}, ErrInvalidSignature
+ 	}
+ 	encoded, sig := parts[0], parts[1]
+ 	expected := m.sign(encoded)
+ 	if !hmac.Equal([]byte(sig), []byte(expected)) {
+ 		return cookiePayload{}, ErrInvalidSignature
+ 	}
+ 	raw, err := base64.RawURLEncoding.DecodeString(encoded)
+ 	if err != nil {
+ 		return cookiePayload{}, fmt.Errorf("session: base64 decode: %w", err)
+ 	}
+ 	var p cookiePayload
+ 	if err := json.Unmarshal(raw, &p); err != nil {
+ 		return cookiePayload{}, fmt.Errorf("session: json unmarshal: %w", err)
+ 	}
+ 	return p, nil
+ }
+ 
+ func (m *Middleware) sign(data string) string {
+ 	mac := hmac.New(sha256.New, m.opts.Secret)
+ 	mac.Write([]byte(data))
+ 	return base64.RawURLEncoding.EncodeToString(mac.Sum(nil))
+ }
+ 
+ // generateID produces a cryptographically random 128-bit session identifier.
+ func generateID() (string, error) {

+ 	if _, err := rand.Read(b); err != nil {
+ 		return "", fmt.Errorf("session: generate id: %w", err)
+ 	}
+ 	return base64.RawURLEncoding.EncodeToString(b), nil
+ }
