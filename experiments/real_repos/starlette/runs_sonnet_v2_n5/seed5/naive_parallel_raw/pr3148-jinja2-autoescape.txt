 def test_templates_with_directories(tmp_path):

     dir2 = tmp_path / "dir2"
     dir2.mkdir()
 
     template1 = dir1 / "index.html"

     templates = Jinja2Templates(directory=[dir1, dir2])
 
     def homepage(request: Request):
         return templates.TemplateResponse(request, "index.html")
 
     app = Starlette(routes=[Route("/", endpoint=homepage)])
     client = TestClient(app)
     response = client.get("/")
