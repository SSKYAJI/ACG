import pytest

from starlette.applications import Starlette
from starlette.middleware.cors import CORSMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse, PlainTextResponse
from starlette.routing import Route
from starlette.testclient import TestClient


def homepage(request: Request):
    return PlainTextResponse("Homepage", status_code=200)


def create_app(**cors_kwargs):
    app = Starlette(routes=[Route("/", homepage)])
    app.add_middleware(CORSMiddleware, **cors_kwargs)
    return app


def test_cors_allow_all_origins():
    app = create_app(allow_origins=["*"])
    client = TestClient(app)

    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "*"


def test_cors_credentials_echoes_origin():
    """When allow_credentials=True and allow_origins=['*'], echo the request Origin."""
    app = create_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app)

    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    # Should echo the explicit origin, not '*'
    assert response.headers["access-control-allow-origin"] == "https://example.com"
    assert response.headers["access-control-allow-credentials"] == "true"


def test_cors_credentials_preflight_echoes_origin():
    """Preflight with allow_credentials=True should echo the request Origin."""
    app = create_app(
        allow_origins=["*"],
        allow_credentials=True,
        allow_methods=["GET", "POST"],
    )
    client = TestClient(app)

    response = client.options(
        "/",
        headers={
            "Origin": "https://example.com",
            "Access-Control-Request-Method": "POST",
        },
    )
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://example.com"
    assert response.headers["access-control-allow-credentials"] == "true"


def test_cors_specific_origin():
    """Specific origin in allow_origins should be echoed."""
    app = create_app(allow_origins=["https://example.com"])
    client = TestClient(app)

    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://example.com"


def test_cors_disallowed_origin():
    """Disallowed origin should get 'null'."""
    app = create_app(allow_origins=["https://allowed.com"])
    client = TestClient(app)

    response = client.get("/", headers={"Origin": "https://evil.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "null"


def test_cors_no_origin_header():
    """Requests without Origin header should pass through without CORS headers."""
    app = create_app(allow_origins=["*"])
    client = TestClient(app)

    response = client.get("/")
    assert response.status_code == 200
    assert "access-control-allow-origin" not in response.headers


def test_cors_preflight_disallowed_method():
    app = create_app(allow_origins=["*"], allow_methods=["GET"])
    client = TestClient(app)

    response = client.options(
        "/",
        headers={
            "Origin": "https://example.com",
            "Access-Control-Request-Method": "DELETE",
        },
    )
    assert response.status_code == 400


def test_cors_wildcard_no_credentials_uses_star():
    """Without credentials, allow_origins=['*'] should return '*'."""
    app = create_app(allow_origins=["*"], allow_credentials=False)
    client = TestClient(app)

    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "*"
    assert "access-control-allow-credentials" not in response.headers


def test_cors_regex_origin():
    """Regex-matched origins should be echoed."""
    app = create_app(allow_origin_regex=r"https://.*\.example\.com")
    client = TestClient(app)

    response = client.get("/", headers={"Origin": "https://sub.example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://sub.example.com"
