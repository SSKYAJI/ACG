+ """
+ SessionMiddleware with read/write distinction.
+
+ Key behaviors:
+ - Sessions are lazily loaded; accessing session data does NOT mark it as modified.
+ - Only explicit mutations (``__setitem__``, ``__delitem__``, ``clear``, ``pop``,
+   ``popitem``, ``update``, ``setdefault``) mark the session as *modified*.
+ - ``Set-Cookie`` is only sent when the session has been modified OR when the
+   session is being cleared (so the client cookie is deleted).
+ - Cookie-clearing behavior (``session.clear()`` followed by response) still
+   sends a ``Set-Cookie`` header that expires the cookie immediately.
+ """

+ # Session store interface

+ class SessionStore:
+     """Minimal in-process session store (replace with Redis/DB in production)."""
+
+     def __init__(self) -> None:
+         self._data: dict[str, dict[str, Any]] = {}
+
+     def load(self, session_id: str) -> dict[str, Any] | None:
+         return self._data.get(session_id)
+
+     def save(self, session_id: str, data: dict[str, Any]) -> None:
+         self._data[session_id] = dict(data)
+
+     def delete(self, session_id: str) -> None:
+         self._data.pop(session_id, None)

+ # ---------------------------------------------------------------------------
+ # Session object

+ class Session(MutableMapping):
+ 	"""
+ 	A dict-like session object that tracks whether it has been *modified*.
+
+ 	Read operations (``__getitem__``, ``__iter__``, ``__len__``, ``get``,
+ 	``keys``, ``values``, ``items``, ``__contains__``) do **not** set the
+ 	modified flag.
+
+ 	Write operations set ``self.modified = True`` so the middleware knows it
+ 	must persist the session and send ``Set-Cookie``.
+
+ 	``self.cleared`` is set to ``True`` when ``clear()`` is called so the
+ 	middleware can expire the client cookie even if the resulting session data
+ 	is empty (and therefore indistinguishable from a brand-new session).
+ 	"""
+
+ 	def __init__(self, session_id: str, data: dict[str, Any] | None = None) -> None:
+ 		self.session_id: str = session_id
+ 		self._data: dict[str, Any] = dict(data) if data else {}
+ 		self.modified: bool = False
+ 		self.cleared: bool = False
+
+ 	# ------------------------------------------------------------------
+ 	# MutableMapping interface
+ 	# ------------------------------------------------------------------
+
+ 	def __getitem__(self, key: str) -> Any:
+ 		return self._data[key]
+
+ 	def __setitem__(self, key: str, value: Any) -> None:
+ 		self._data[key] = value
+ 		self.modified = True
+
+ 	def __delitem__(self, key: str) -> None:
+ 		del self._data[key]
+ 		self.modified = True
+
+ 	def __iter__(self) -> Iterator:
+ 		return iter(self._data)
+
+ 	def __len__(self) -> int:
+ 		return len(self._data)
+
+ 	def __contains__(self, key: object) -> bool:
+ 		return key in self._data
+
+ 	# ------------------------------------------------------------------
+ 	# Override mutating helpers so they set the flag too
+ 	# ------------------------------------------------------------------
+
+ 	def clear(self) -> None:  # type: ignore[override]
+ 		self._data.clear()
+ 		self.modified = True
+ 		self.cleared = True
+
+ 	def pop(self, key: str, *args: Any) -> Any:
+ 		result = self._data.pop(key, *args)
+ 		self.modified = True
+ 		return result
+
+ 	def popitem(self) -> tuple[str, Any]:
+ 		result = self._data.popitem()
+ 		self.modified = True
+ 		return result
+
+ 	def update(self, other: Any = (), **kwargs: Any) -> None:  # type: ignore[override]
+ 		self._data.update(other, **kwargs)
+ 		self.modified = True
+
+ 	def setdefault(self, key: str, default: Any = None) -> Any:  # type: ignore[override]
+ 		if key not in self._data:
+ 			self._data[key] = default
+ 			self.modified = True
+ 		return self._data[key]
+
+ 	# ------------------------------------------------------------------
+ 	# Convenience
+ 	# ------------------------------------------------------------------
+
+ 	def __repr__(self) -> str:
+ 		return (
+ 			f"Session(id={self.session_id!r}, modified={self.modified}, "
+ 			f"cleared={self.cleared}, data={self._data!r})"

+ class Request:
+     """Minimal request wrapper used by the middleware."""
+
+     def __init__(self, environ: dict[str, Any]) -> None:
+         self.environ = environ
+         self._cookies: dict[str, str] | None = None
+
+     @property
+     def cookies(self) -> dict[str, str]:
+         if self._cookies is None:
+             self._cookies = {}
+             cookie_header = self.environ.get("HTTP_COOKIE", "")
+             if cookie_header:
+                 morsel_map = http.cookies.SimpleCookie()
+                 morsel_map.load(cookie_header)
+                 self._cookies = {k: v.value for k, v in morsel_map.items()}
+         return self._cookies
+

+ class Response:
+     """Minimal response object."""
+
+     def __init__(
+         self,
+         body: bytes = b"",
+         status: int = 200,
+         headers: dict[str, str] | None = None,
+     ) -> None:
+         self.body = body
+         self.status = status
+         self.headers: dict[str, str] = headers or {}
+         self._extra_headers: list[tuple[str, str]] = []
+
+     def set_cookie(
+         self,
+         name: str,
+         value: str,
+         *,
+         max_age: int | None = None,
+         httponly: bool = True,
+         samesite: str = "Lax",
+         secure: bool = False,
+         path: str = "/",
+     ) -> None:
+         cookie: http.cookies.SimpleCookie = http.cookies.SimpleCookie()
+         cookie[name] = value
+         m = cookie[name]
+         m["path"] = path
+         m["httponly"] = httponly
+         m["samesite"] = samesite
+         if secure:
+             m["secure"] = secure
+         if max_age is not None:
+             m["max-age"] = max_age
+         # Store as extra header so multiple Set-Cookie headers are possible.
+         self._extra_headers.append(("Set-Cookie", m.OutputString()))
+
+     def delete_cookie(self, name: str, path: str = "/") -> None:
+         self.set_cookie(name, "", max_age=0, path=path)
+
+     def get_headers(self) -> list[tuple[str, str]]:
+         headers = list(self.headers.items()) + self._extra_headers
+         return headers

+ class SessionMiddleware:
+     """
+     Wraps a handler and provides transparent session management.
+
+     Parameters
+     ----------
+     handler:
+         The inner WSGI-style handler ``(Request) -> Response``.
+     store:
+         A :class:`SessionStore` instance (or compatible object).
+     cookie_name:
+         Name of the session cookie sent to the browser.
+     cookie_max_age:
+         Lifetime of the session cookie in seconds.  ``None`` means a
+         session cookie (expires when the browser closes).
+     cookie_secure:
+         Set the ``Secure`` flag on the cookie.
+     cookie_httponly:
+         Set the ``HttpOnly`` flag on the cookie.
+     cookie_samesite:
+         Value for the ``SameSite`` attribute.
+     """
+
+     def __init__(
+         self,
+         handler: HandlerType,
+         store: SessionStore,
+         *,
+         cookie_name: str = "session_id",
+         cookie_max_age: int | None = 3600,
+         cookie_secure: bool = False,
+         cookie_httponly: bool = True,
+         cookie_samesite: str = "Lax",
+     ) -> None:
+         self.handler = handler
+         self.store = store
+         self.cookie_name = cookie_name
+         self.cookie_max_age = cookie_max_age
+         self.cookie_secure = cookie_secure
+         self.cookie_httponly = cookie_httponly
+         self.cookie_samesite = cookie_samesite
+

+     # Public entry point
+     # ------------------------------------------------------------------
+
+     def __call__(self, request: Request) -> Response:
+         session = self._load_session(request)
+         # Attach session to request so the handler can access it.
+         request.environ["session"] = session
+
+         response = self.handler(request)
+
+         self._process_session(session, response)
+         return response
+
+     # ------------------------------------------------------------------
+     # Internal helpers
+     # ------------------------------------------------------------------
+
+     def _load_session(self, request: Request) -> Session:
+         session_id = request.cookies.get(self.cookie_name)
+         if session_id:

+         new_id = str(uuid.uuid4())

+
+     def _process_session(self, session: Session, response: Response) -> None:
+         """
+         Decide whether to persist the session and/or send ``Set-Cookie``.
+
+         Rules
+         -----
+         1. If the session was **cleared**, delete it from the store and send
+            an expiring ``Set-Cookie`` so the browser removes the cookie.
+         2. If the session was **modified** (but not cleared), save it to the
+            store and send a fresh ``Set-Cookie``.
+         3. If the session was **not modified**, do nothing – no ``Set-Cookie``
+            header is added to the response.

+                 httponly=self.cookie_httponly,
+                 samesite=self.cookie_samesite,
+                 secure=self.cookie_secure,
