import pytest

from starlette.applications import Starlette
from starlette.middleware.cors import CORSMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse
from starlette.routing import Route
from starlette.testclient import TestClient


def homepage(request: Request):
    return JSONResponse({"hello": "world"})


def build_app(**cors_kwargs):
    app = Starlette(routes=[Route("/", homepage)])
    app.add_middleware(CORSMiddleware, **cors_kwargs)
    return app


# ---------------------------------------------------------------------------
# Basic wildcard
# ---------------------------------------------------------------------------


def test_cors_allow_all_origins():
    app = build_app(allow_origins=["*"])
    client = TestClient(app)
    r = client.get("/", headers={"Origin": "https://example.com"})
    assert r.headers["Access-Control-Allow-Origin"] == "*"


# ---------------------------------------------------------------------------
# Credentials + wildcard: must echo origin, not '*'
# ---------------------------------------------------------------------------


def test_cors_credentials_with_wildcard_echoes_origin():
    """
    When allow_origins=['*'] and allow_credentials=True, the simple response
    must echo the request Origin rather than returning '*', because browsers
    reject credentials with a wildcard origin.
    """
    app = build_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app)
    r = client.get("/", headers={"Origin": "https://trusted.example.com"})
    assert r.headers["Access-Control-Allow-Origin"] == "https://trusted.example.com"
    assert r.headers.get("Access-Control-Allow-Credentials") == "true"


def test_cors_credentials_preflight_echoes_origin():
    """Preflight with credentials should also echo the origin."""
    app = build_app(
        allow_origins=["*"],
        allow_credentials=True,
        allow_methods=["GET", "POST"],
    )
    client = TestClient(app)
    r = client.options(
        "/",
        headers={
            "Origin": "https://trusted.example.com",
            "Access-Control-Request-Method": "POST",
        },
    )
    assert r.status_code == 200
    assert r.headers["Access-Control-Allow-Origin"] == "https://trusted.example.com"
    assert r.headers.get("Access-Control-Allow-Credentials") == "true"


# ---------------------------------------------------------------------------
# Specific origins (no wildcard)
# ---------------------------------------------------------------------------


def test_cors_specific_origin_allowed():
    app = build_app(allow_origins=["https://example.com"])
    client = TestClient(app)
    r = client.get("/", headers={"Origin": "https://example.com"})
    assert r.headers["Access-Control-Allow-Origin"] == "https://example.com"


def test_cors_specific_origin_disallowed():
    app = build_app(allow_origins=["https://example.com"])
    client = TestClient(app)
    r = client.get("/", headers={"Origin": "https://evil.com"})
    assert r.headers.get("Access-Control-Allow-Origin") == "null"


# ---------------------------------------------------------------------------
# No origin header — middleware is transparent
# ---------------------------------------------------------------------------


def test_cors_no_origin_header():
    app = build_app(allow_origins=["*"])
    client = TestClient(app)
    r = client.get("/")
    assert "Access-Control-Allow-Origin" not in r.headers


# ---------------------------------------------------------------------------
# Preflight disallowed method
# ---------------------------------------------------------------------------


def test_cors_preflight_disallowed_method():
    app = build_app(allow_origins=["*"], allow_methods=["GET"])
    client = TestClient(app)
    r = client.options(
        "/",
        headers={
            "Origin": "https://example.com",
            "Access-Control-Request-Method": "DELETE",
        },
    )
    assert r.status_code == 400


# ---------------------------------------------------------------------------
# Vary header
# ---------------------------------------------------------------------------


def test_cors_vary_header_present_when_echoing_origin():
    app = build_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app)
    r = client.get("/", headers={"Origin": "https://example.com"})
    assert "Vary" in r.headers
    assert "Origin" in r.headers["Vary"]
