+ """
+ SessionMiddleware with read/write distinction.
+
+ Key behaviors:
+ - Sessions are lazily loaded; accessing session data does NOT mark it as modified.
+ - Only explicit mutation (``__setitem__``, ``__delitem__``, ``clear``, ``pop``,
+   ``popitem``, ``update``, ``setdefault``) marks the session as *modified*.
+ - If the session was not modified during the request, no ``Set-Cookie`` header is
+   sent (avoids unnecessary cookie churn).
+ - Clearing / deleting a session (``session.clear()`` after it existed, or
+   ``request.session = None``) still sends the appropriate ``Set-Cookie`` header
+   to expire the cookie in the browser.
+ """

+ # ---------------------------------------------------------------------------
+ # Session store (in-memory, swappable)
+ # ---------------------------------------------------------------------------
+
+ class SessionStore:
+     """Thread-safe in-memory session store."""
+
+     def __init__(self) -> None:
+         self._lock = threading.Lock()
+         self._sessions: Dict[str, Dict[str, Any]] = {}
+         self._expiry: Dict[str, float] = {}
+
+     def load(self, session_id: str) -> Optional[Dict[str, Any]]:
+         with self._lock:
+             exp = self._expiry.get(session_id)
+             if exp is not None and time.time() > exp:
+                 self._sessions.pop(session_id, None)
+                 self._expiry.pop(session_id, None)
+                 return None
+             data = self._sessions.get(session_id)
+             return dict(data) if data is not None else None
+
+     def save(self, session_id: str, data: Dict[str, Any], max_age: int) -> None:
+         with self._lock:
+             self._sessions[session_id] = dict(data)
+             self._expiry[session_id] = time.time() + max_age
+
+     def delete(self, session_id: str) -> None:
+         with self._lock:
+             self._sessions.pop(session_id, None)
+             self._expiry.pop(session_id, None)
+
+     def exists(self, session_id: str) -> bool:
+         with self._lock:
+             return session_id in self._sessions

+         # whether a clear() should trigger cookie expiry.

+     # MutableMapping interface

+
+     def __delitem__(self, key: str) -> None:
+         del self._data[key]

+
+     def __iter__(self) -> Iterator:

+     def __len__(self) -> int:

+ # ---------------------------------------------------------------------------
+ # Signer (HMAC-SHA256 for session IDs)
+ # ---------------------------------------------------------------------------
+
+ class Signer:
+     """Simple HMAC-SHA256 signer for session IDs stored in cookies."""
+
+     SEP = "."
+
+     def __init__(self, secret: str) -> None:
+         self._secret = secret.encode()
+
+     def sign(self, value: str) -> str:
+         sig = hmac.new(self._secret, value.encode(), hashlib.sha256).hexdigest()
+         return f"{value}{self.SEP}{sig}"
+
+     def unsign(self, signed_value: str) -> Optional[str]:
+         if self.SEP not in signed_value:
+             return None
+         value, sig = signed_value.rsplit(self.SEP, 1)
+         expected = hmac.new(self._secret, value.encode(), hashlib.sha256).hexdigest()
+         if hmac.compare_digest(sig, expected):
+             return value
+         return None
