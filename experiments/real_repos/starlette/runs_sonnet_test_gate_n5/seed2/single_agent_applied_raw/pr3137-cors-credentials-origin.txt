import pytest
from starlette.applications import Starlette
from starlette.middleware.cors import CORSMiddleware
from starlette.requests import Request
from starlette.responses import PlainTextResponse
from starlette.routing import Route
from starlette.testclient import TestClient


def homepage(request: Request):
    return PlainTextResponse("Homepage", status_code=200)


def create_app(allow_origins, allow_credentials=False, **kwargs):
    app = Starlette(routes=[Route("/", homepage)])
    app = CORSMiddleware(
        app,
        allow_origins=allow_origins,
        allow_credentials=allow_credentials,
        allow_methods=["GET", "POST"],
        **kwargs,
    )
    return app


def test_simple_request_with_wildcard_no_credentials():
    """With allow_origins=['*'] and no credentials, response should have '*'."""
    app = create_app(allow_origins=["*"], allow_credentials=False)
    client = TestClient(app, raise_server_exceptions=False)

    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    # With wildcard and no credentials, the origin header should be '*' or echoed
    assert "access-control-allow-origin" in response.headers


def test_simple_request_with_wildcard_and_credentials():
    """
    With allow_origins=['*'] and allow_credentials=True,
    the response should echo the explicit request Origin instead of '*'.
    """
    app = create_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app, raise_server_exceptions=False)

    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    # Must echo the origin, not '*', when credentials are allowed
    assert response.headers["access-control-allow-origin"] == "https://example.com"
    assert response.headers.get("access-control-allow-credentials") == "true"


def test_simple_request_with_wildcard_and_credentials_different_origin():
    """Echo the correct origin for different origins."""
    app = create_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app, raise_server_exceptions=False)

    response = client.get("/", headers={"Origin": "https://other.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://other.com"


def test_preflight_with_wildcard_and_credentials():
    """
    Preflight with allow_origins=['*'] and allow_credentials=True
    should echo the explicit request Origin.
    """
    app = create_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app, raise_server_exceptions=False)

    response = client.options(
        "/",
        headers={
            "Origin": "https://example.com",
            "Access-Control-Request-Method": "GET",
        },
    )
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://example.com"
    assert response.headers.get("access-control-allow-credentials") == "true"


def test_preflight_with_explicit_origin():
    """Preflight with explicit origin list should echo the origin."""
    app = create_app(
        allow_origins=["https://example.com"], allow_credentials=False

    client = TestClient(app, raise_server_exceptions=False)

    assert response.headers["access-control-allow-origin"] == "https://example.com"


def test_simple_request_disallowed_origin():
    """Requests from disallowed origins should get 'null' or no ACAO header."""
    app = create_app(allow_origins=["https://allowed.com"], allow_credentials=False)
    client = TestClient(app, raise_server_exceptions=False)

    response = client.get("/", headers={"Origin": "https://evil.com"})
    assert response.status_code == 200
    # Origin should be 'null' for disallowed origins
    assert response.headers.get("access-control-allow-origin") == "null"


def test_vary_header_set_when_echoing_origin():
    """When echoing origin, Vary: Origin should be set."""
    app = create_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app, raise_server_exceptions=False)

    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    # Vary header should include Origin when we echo the origin
    vary = response.headers.get("vary", "")
    assert "Origin" in vary
