+ // MemoryStore is a simple in-memory session store suitable for testing.
+ type MemoryStore struct {
+ 	mu      sync.RWMutex
+ 	entries map[string]memEntry
+ }
+
+ type memEntry struct {
+ 	data    map[string]interface{}
+ 	expires time.Time
+ }
+
+ // NewMemoryStore creates a new MemoryStore.
+ func NewMemoryStore() *MemoryStore {
+ 	return &MemoryStore{
+ 		entries: make(map[string]memEntry),
+ 	}
+ }
+
+ // Load retrieves session data. Returns nil, nil if not found or expired.
+ func (s *MemoryStore) Load(_ context.Context, sessionID string) (map[string]interface{}, error) {
+ 	s.mu.RLock()
+ 	defer s.mu.RUnlock()
+ 	e, ok := s.entries[sessionID]
+ 	if !ok {
+ 		return nil, nil
+ 	}
+ 	if time.Now().After(e.expires) {
+ 		return nil, nil
+ 	}
+ 	return deepCopy(e.data), nil
+ }
+
+ // Save stores session data with the given TTL.
+ func (s *MemoryStore) Save(_ context.Context, sessionID string, data map[string]interface{}, ttl time.Duration) error {
+ 	s.mu.Lock()
+ 	defer s.mu.Unlock()
+ 	s.entries[sessionID] = memEntry{
+ 		data:    deepCopy(data),
+ 		expires: time.Now().Add(ttl),
+ 	}
+ 	return nil
+ }
+
+ // Delete removes a session.
+ func (s *MemoryStore) Delete(_ context.Context, sessionID string) error {
+ 	s.mu.Lock()
+ 	defer s.mu.Unlock()
+ 	delete(s.entries, sessionID)
+ 	return nil
+ }
+
+ // Has reports whether a session ID exists and is not expired.
+ func (s *MemoryStore) Has(sessionID string) bool {
+ 	s.mu.RLock()
+ 	defer s.mu.RUnlock()
+ 	e, ok := s.entries[sessionID]
+ 	if !ok {
+ 		return false
+ 	}
