+ class Response:
+     media_type = None
+     charset = "utf-8"
+ 
+     def __init__(
+         self,
+         content: typing.Any = None,
+         status_code: int = 200,
+         headers: dict = None,  # type: ignore[assignment]
+         media_type: str = None,  # type: ignore[assignment]
+         background=None,  # type: ignore[no-untyped-def]
+     ) -> None:
+         self.status_code = status_code
+         if headers is None:
+             raw_headers: typing.List[typing.Tuple[bytes, bytes]] = []
+             populate_content_length = True
+             populate_content_type = True
+         else:
+             raw_headers = [
+                 (k.lower().encode("latin-1"), v.encode("latin-1"))
+                 for k, v in headers.items()
+             ]
+             keys = [h[0] for h in raw_headers]
+             populate_content_length = b"content-length" not in keys
+             populate_content_type = b"content-type" not in keys
+ 
+         body = self.render(content)
+         if body is not None:
+             if populate_content_length:
+                 content_length = str(len(body))
+                 raw_headers.append((b"content-length", content_length.encode("latin-1")))
+             if populate_content_type:
+                 content_type = self.media_type or "text/plain"
+                 if content_type.startswith("text/") and "charset=" not in content_type:
+                     content_type += "; charset=" + self.charset
+                 raw_headers.append((b"content-type", content_type.encode("latin-1")))
+ 
+         self.body = body
+         self.background = background
+         self.raw_headers = raw_headers
+ 
+     def render(self, content: typing.Any) -> typing.Optional[bytes]:
+         if content is None:
+             return None
+         if isinstance(content, bytes):
+             return content
+         return content.encode(self.charset)
+ 
+     async def __call__(self, scope, receive, send):  # type: ignore[no-untyped-def]
+         prefix = "websocket" if scope["type"] == "websocket" else "http"
+         await send(
+             {
+                 "type": prefix + ".response.start",
+                 "status": self.status_code,
+                 "headers": self.raw_headers,
+             }
+         )
+         await send({"type": prefix + ".response.body", "body": self.body or b""})
+         if self.background is not None:
+             await self.background()
