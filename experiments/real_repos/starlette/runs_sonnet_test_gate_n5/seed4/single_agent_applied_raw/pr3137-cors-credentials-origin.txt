import pytest

from starlette.applications import Starlette
from starlette.middleware.cors import CORSMiddleware
from starlette.requests import Request
from starlette.responses import JSONResponse
from starlette.routing import Route
from starlette.testclient import TestClient


def create_app(
    allow_origins=("*",),
    allow_credentials=False,
    allow_methods=("GET",),
    allow_headers=(),
    allow_origin_regex=None,
    expose_headers=(),
):
    async def homepage(request: Request):
        return JSONResponse({"hello": "world"})

    app = Starlette(routes=[Route("/", homepage)])
    app.add_middleware(
        CORSMiddleware,
        allow_origins=allow_origins,
        allow_credentials=allow_credentials,
        allow_methods=allow_methods,
        allow_headers=allow_headers,
        allow_origin_regex=allow_origin_regex,
        expose_headers=expose_headers,
    )
    return app


def test_cors_allow_all_origins():
    """With allow_origins=['*'] and no credentials, wildcard is returned."""
    app = create_app(allow_origins=["*"])
    client = TestClient(app)
    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "*"


def test_cors_credentials_echoes_origin():
    """With allow_origins=['*'] and allow_credentials=True, the actual origin is echoed."""
    app = create_app(allow_origins=["*"], allow_credentials=True)
    client = TestClient(app)
    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    # Should echo the actual origin, not '*'
    assert response.headers["access-control-allow-origin"] == "https://example.com"
    assert response.headers.get("access-control-allow-credentials") == "true"
    assert response.headers.get("vary") == "Origin"


def test_cors_preflight_credentials_echoes_origin():
    """Preflight with allow_origins=['*'] and allow_credentials=True echoes origin."""
    app = create_app(
        allow_origins=["*"],
        allow_credentials=True,
        allow_methods=["GET", "POST"],
    )
    client = TestClient(app)
    response = client.options(
        "/",
        headers={
            "Origin": "https://example.com",
            "Access-Control-Request-Method": "GET",
        },
    )
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://example.com"
    assert response.headers.get("access-control-allow-credentials") == "true"


def test_cors_specific_origin():
    """With specific allowed origins, the origin is echoed."""
    app = create_app(allow_origins=["https://example.com"])
    client = TestClient(app)
    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "https://example.com"


def test_cors_disallowed_origin():
    """With specific allowed origins, disallowed origins get 'null'."""
    app = create_app(allow_origins=["https://example.com"])
    client = TestClient(app)
    response = client.get("/", headers={"Origin": "https://other.com"})
    assert response.status_code == 200
    assert response.headers["access-control-allow-origin"] == "null"


def test_cors_no_origin_header():
    """Requests without Origin header pass through without CORS headers."""
    app = create_app(allow_origins=["*"])
    client = TestClient(app)
    response = client.get("/")
    assert response.status_code == 200
    assert "access-control-allow-origin" not in response.headers


def test_cors_preflight_disallowed_method():
    """Preflight with disallowed method returns 400."""
    app = create_app(allow_origins=["*"], allow_methods=["GET"])
    client = TestClient(app)
    response = client.options(
        "/",
        headers={
            "Origin": "https://example.com",
            "Access-Control-Request-Method": "DELETE",
        },
    )
    assert response.status_code == 400


def test_cors_expose_headers():
    """Expose-Headers should be included in simple responses."""
    app = create_app(allow_origins=["*"], expose_headers=["X-Custom-Header"])
    client = TestClient(app)
    response = client.get("/", headers={"Origin": "https://example.com"})
    assert response.status_code == 200
    assert "X-Custom-Header" in response.headers.get("access-control-expose-headers", "")
