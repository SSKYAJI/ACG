+ class BackgroundTask:
+     def __init__(
+         self, func: typing.Callable, *args: typing.Any, **kwargs: typing.Any
+     ) -> None:
+         self.func = func
+         self.args = args
+         self.kwargs = kwargs
+         self.is_async = asyncio.iscoroutinefunction(func)
+ 
+     async def __call__(self) -> None:
+         if self.is_async:
+             await self.func(*self.args, **self.kwargs)
+         else:
+             await asyncio.get_event_loop().run_in_executor(
+                 None, self.func, *self.args, **self.kwargs
+             )

+         task = BackgroundTask(func, *args, **kwargs)
+         self.tasks.append(task)

+ class HTTPConnection(typing.Mapping[str, typing.Any]):
+     """
+     A base class for incoming HTTP connections.
+     """
+ 
+     def __init__(self, scope: dict) -> None:
+         assert scope["type"] in ("http", "websocket")
+         self._scope = scope
+ 
+     def __getitem__(self, key: str) -> str:
+         return self._scope[key]
+ 
+     def __iter__(self) -> typing.Iterator[str]:
+         return iter(self._scope)
+ 
+     def __len__(self) -> int:
+         return len(self._scope)
+ 
+     @property
+     def method(self) -> str:
+         return self._scope["method"]
+ 
+     @property
+     def url(self) -> URL:
+         if not hasattr(self, "_url"):
+             scheme = self._scope["scheme"] if "scheme" in self._scope else "http"
+             server = self._scope.get("server")
+             path = self._scope.get("root_path", "") + self._scope["path"]
+             if server is None:
+                 url = path
+             else:
+                 host, port = server
+                 default_port = {"http": 80, "https": 443, "ws": 80, "wss": 443}[scheme]
+                 if port == default_port:
+                     url = f"{scheme}://{host}{path}"
+                 else:
+                     url = f"{scheme}://{host}:{port}{path}"
+             query_string = self._scope.get("query_string", b"")
+             if query_string:
+                 url += "?" + query_string.decode()
+             self._url = URL(url)
+         return self._url
+ 
+     @property
+     def headers(self) -> Headers:
+         if not hasattr(self, "_headers"):
+             self._headers = Headers(scope=self._scope)

+         endpoint: typing.Callable,

+         methods: typing.List[str] = None,
+         name: str = None,

+         self.path = path
+         self.endpoint = endpoint
+         self.methods = methods
+         self.name = name or endpoint.__name__
+ 
+     async def handle(self, scope, receive, send):
+         request = Request(scope, receive)
+         if asyncio.iscoroutinefunction(self.endpoint):
+             response = await self.endpoint(request)
+         else:
+             response = self.endpoint(request)
+         await response(scope, receive, send)
+ 
+     def matches(self, scope):
+         if scope["type"] == "http":
+             path = scope["path"]
+             if path == self.path:
+                 if self.methods is None or scope["method"] in self.methods:
+                     return True
+         return False

+         self,

+     ) -> None:
+         self.routes = [] if routes is None else routes
+ 
+     async def __call__(self, scope, receive, send) -> None:
+         scope["app"] = self
+         if scope["type"] == "http":
+             for route in self.routes:
+                 if route.matches(scope):
+                     await route.handle(scope, receive, send)

+             # 404
