+ SessionMiddleware with read/write distinction.
+
+ Key behaviors:
+ - Sessions are lazily loaded; accessing session data does NOT mark it as modified.
+ - Only explicit mutation (``__setitem__``, ``__delitem__``, ``clear``, ``pop``,
+   ``update``, ``setdefault``) marks the session as *modified*.
+ - ``Set-Cookie`` is only sent when the session is modified OR when the session
+   is being cleared (so that the cookie is deleted from the browser).
+ - Cookie clearing (``session.clear()`` followed by save, or explicit
+   ``session.delete()``) still works correctly.
+ """

+ class SessionStore:
+     """Very small in-process session store (for demonstration / testing)."""
+ 
+     def __init__(self) -> None:
+         self._store: dict[str, dict[str, Any]] = {}
+ 
+     def load(self, session_id: str) -> Optional[dict[str, Any]]:
+         return self._store.get(session_id)
+ 
+     def save(self, session_id: str, data: dict[str, Any]) -> None:
+         self._store[session_id] = dict(data)
+ 
+     def delete(self, session_id: str) -> None:
+         self._store.pop(session_id, None)
+ 
+     def exists(self, session_id: str) -> bool:

+ _COOKIE_NAME = "session_id"
+ _SECRET_KEY = os.environ.get("SESSION_SECRET", "change-me-in-production")

+ def _sign(value: str, secret: str = _SECRET_KEY) -> str:
+     sig = hmac.new(secret.encode(), value.encode(), hashlib.sha256).hexdigest()
+     return f"{value}.{sig}"
+ 
+ 
+ def _unsign(signed: str, secret: str = _SECRET_KEY) -> Optional[str]:
+     if "." not in signed:
+         return None
+     value, _, sig = signed.rpartition(".")
+     expected = hmac.new(secret.encode(), value.encode(), hashlib.sha256).hexdigest()
+     if hmac.compare_digest(sig, expected):
+         return value
+     return None

+ class SessionMiddleware:

+ 
+     ``Set-Cookie`` is emitted **only** when:
+     * The session was modified (data changed), OR

+     """

+         self.cookie_name = cookie_name
+         self.secret = secret
+         self.max_age = max_age
+         self.secure = secure
+         self.httponly = httponly
+         self.samesite = samesite

+         cookie_header = environ.get("HTTP_COOKIE", "")
+         if not cookie_header:
+             return None
+         cookie: SimpleCookie = SimpleCookie()
+         try:
+             cookie.load(cookie_header)
+         except Exception:
+             return None
+         morsel = cookie.get(self.cookie_name)
+         if morsel is None:
+             return None
+         return _unsign(morsel.value, self.secret)
