+ type entry struct {
+ 	data    map[string]interface{}
+ 	expires time.Time
+ }
+
+ // MemoryStore is an in-memory session store suitable for testing.
+ type MemoryStore struct {
+ 	mu      sync.RWMutex
+ 	entries map[string]entry
+ }
+
+ // NewMemoryStore creates a new MemoryStore.
+ func NewMemoryStore() *MemoryStore {
+ 	return &MemoryStore{
+ 		entries: make(map[string]entry),
+ 	}
+ }
+
+ // Load retrieves session data by ID.
+ func (m *MemoryStore) Load(id string) (map[string]interface{}, error) {
+ 	m.mu.RLock()
+ 	defer m.mu.RUnlock()
+ 	e, ok := m.entries[id]
+ 	if !ok {
+ 		return nil, nil
+ 	}
+ 	if !e.expires.IsZero() && time.Now().After(e.expires) {
+ 		return nil, nil
+ 	}
+ 	// Return a copy to avoid data races.
+ 	snapshot := make(map[string]interface{}, len(e.data))
+ 	for k, v := range e.data {
+ 		snapshot[k] = v
+ 	}
+ 	return snapshot, nil
+ }
+
+ // Save persists session data with an optional TTL (zero means no expiry).
+ func (m *MemoryStore) Save(id string, data map[string]interface{}, ttl time.Duration) error {
+ 	m.mu.Lock()
+ 	defer m.mu.Unlock()
+ 	snapshot := make(map[string]interface{}, len(data))
+ 	for k, v := range data {
+ 		snapshot[k] = v
+ 	}
+ 	var expires time.Time
+ 	if ttl > 0 {
+ 		expires = time.Now().Add(ttl)
+ 	}
+ 	m.entries[id] = entry{data: snapshot, expires: expires}
+ 	return nil
+ }
+
+ // Delete removes a session from the store.
+ func (m *MemoryStore) Delete(id string) error {
+ 	m.mu.Lock()
+ 	defer m.mu.Unlock()
+ 	delete(m.entries, id)
+ 	return nil
+ }

+ package session
+
+ import (
+ 	"context"
+ 	"net/http"
+ 	"time"
+ )
+
+ type contextKey struct{}
+
+ // Options configures the SessionMiddleware.
+ type Options struct {
+ 	// CookieName is the name of the session cookie. Defaults to "session_id".
+ 	CookieName string
+ 	// TTL is the session lifetime. Zero means session cookie (browser lifetime).
+ 	TTL time.Duration
+ 	// CookiePath defaults to "/".
+ 	CookiePath string
+ 	// CookieSecure sets the Secure flag on the cookie.
+ 	CookieSecure bool
+ 	// CookieHTTPOnly sets the HttpOnly flag on the cookie.
+ 	CookieHTTPOnly bool
+ 	// CookieSameSite sets the SameSite attribute.
+ 	CookieSameSite http.SameSite
+ 	// Store is the session backend. Defaults to a new MemoryStore if nil.
+ 	Store Store
+ }
+
+ func (o *Options) cookieName() string {
+ 	if o.CookieName == "" {
+ 		return "session_id"
+ 	}
+ 	return o.CookieName
+ }
+
+ func (o *Options) cookiePath() string {
+ 	if o.CookiePath == "" {
+ 		return "/"
+ 	}
+ 	return o.CookiePath
+ }
+
+ // Middleware returns an HTTP middleware that manages sessions.
+ //
+ // Behaviour:
+ //   - If the request carries a valid session cookie the session is loaded from
+ //     the store; otherwise a new session is created lazily (the cookie is only
+ //     sent when the session is actually mutated).
+ //   - Set-Cookie is only emitted when the session has been mutated.
+ //   - When Session.Clear() is called the cookie is actively expired (cleared).
+ //   - The Session is stored in the request context and can be retrieved with
+ //     FromContext.
+ func Middleware(opts Options) func(http.Handler) http.Handler {
+ 	if opts.Store == nil {
+ 		opts.Store = NewMemoryStore()
+ 	}
+ 	return func(next http.Handler) http.Handler {
+ 		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
+ 			sess, isNew := loadOrCreate(r, opts)
+
+ 			// Inject session into context.
+ 			ctx := context.WithValue(r.Context(), contextKey{}, sess)
+ 			r = r.WithContext(ctx)
+
+ 			next.ServeHTTP(w, r)
+
+ 			// Post-handler: decide whether to persist / clear.
+ 			if sess.IsCleared() {
+ 				// Delete from store and expire the cookie.
+ 				_ = opts.Store.Delete(sess.ID())
+ 				clearCookie(w, opts)
+ 				return
+ 			}
+
+ 			if sess.IsMutated() {
+ 				// Persist to store.
+ 				_ = opts.Store.Save(sess.ID(), sess.Data(), opts.TTL)
+ 				// Only set the cookie when the session is new or mutated.
+ 				if isNew || sess.IsMutated() {
+ 					setCookie(w, sess.ID(), opts)
+ 				}
+ 			}
+ 			// If not mutated: do nothing – no Set-Cookie header is emitted.
+ 		})
+ 	}
+ }
+
+ // FromContext retrieves the Session from the context. Returns nil if not present.
+ func FromContext(ctx context.Context) *Session {
+ 	s, _ := ctx.Value(contextKey{}).(*Session)
+ 	return s
+ }
+
+ // loadOrCreate loads an existing session from the store or creates a new one.
+ // isNew is true when no valid cookie was found.
+ func loadOrCreate(r *http.Request, opts Options) (sess *Session, isNew bool) {
+ 	cookie, err := r.Cookie(opts.cookieName())
+ 	if err == nil && cookie.Value != "" {
+ 		data, loadErr := opts.Store.Load(cookie.Value)
+ 		if loadErr == nil && data != nil {
+ 			return LoadSession(cookie.Value, data), false
+ 		}
+ 	}
+ 	// Create a new session (ID generated eagerly so it is stable).
+ 	sess, _ = NewSession()
+ 	return sess, true
+ }
+
+ func setCookie(w http.ResponseWriter, id string, opts Options) {
+ 	c := &http.Cookie{
+ 		Name:     opts.cookieName(),
+ 		Value:    id,
+ 		Path:     opts.cookiePath(),
+ 		Secure:   opts.CookieSecure,
+ 		HttpOnly: opts.CookieHTTPOnly,
+ 		SameSite: opts.CookieSameSite,
+ 	}
+ 	if opts.TTL > 0 {
+ 		c.MaxAge = int(opts.TTL.Seconds())
+ 	}
+ 	http.SetCookie(w, c)
+ }
+
+ func clearCookie(w http.ResponseWriter, opts Options) {
+ 	http.SetCookie(w, &http.Cookie{
+ 		Name:     opts.cookieName(),
+ 		Value:    "",
+ 		Path:     opts.cookiePath(),
+ 		MaxAge:   -1,
+ 		Secure:   opts.CookieSecure,
+ 		HttpOnly: opts.CookieHTTPOnly,
+ 	})
+ }
