+ def test_cors_allow_all_origins_with_credentials_echoes_origin():
+     """
+     When allow_origins=['*'] and allow_credentials=True, simple responses
+     must echo the explicit request Origin header instead of '*'.
+     """
+     client = TestClient(
+         CORSMiddleware(
+             app,
+             allow_origins=["*"],

+         ),
+         raise_server_exceptions=True,
+     )
+     response = client.get("/", headers={"Origin": "https://example.com"})
