@pytest.mark.skipif(jinja2 is None, reason="jinja2 not installed")
def test_template_with_autoescape_enabled(tmpdir, test_client_factory):
    """Test that HTML templates are autoescaped by default."""
    path = os.path.join(tmpdir, "index.html")
    with open(path, "w") as f:
        f.write("<html>{{ dangerous }}</html>")

    async def homepage(request):
        return templates.TemplateResponse(
            "index.html", {"request": request, "dangerous": "<script>alert('xss')</script>"}
        )

    app = Starlette(
        debug=True,
        routes=[Route("/", endpoint=homepage)],
    )
    templates = Jinja2Templates(directory=str(tmpdir))

    client = test_client_factory(app)
    response = client.get("/")
    assert "<script>" not in response.text
    assert "&lt;script&gt;" in response.text
