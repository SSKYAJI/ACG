+def test_session_read_only_no_set_cookie():
+    """
+    When a request only reads the session without modifying it,
+    the response should not include a Set-Cookie header.
+    """
+    app = Starlette()
+    app.add_middleware(SessionMiddleware, secret_key="test-secret")
+
+    @app.route("/")
+    async def homepage(request):
+        # Access session but do not modify
+        session = request.session
+        _ = session.get("key", "default")
+        return JSONResponse({"session": dict(session)})
+
+    client = TestClient(app)
+    # First request to set a session cookie
+    response = client.get("/", cookies={"session": "eyJrZXkiOiJ2YWx1ZSJ9.ZmFrZQ==.signature"})
+    # The session is invalid, so it will be empty; but we are not modifying, so no Set-Cookie
+    assert "Set-Cookie" not in response.headers
+
