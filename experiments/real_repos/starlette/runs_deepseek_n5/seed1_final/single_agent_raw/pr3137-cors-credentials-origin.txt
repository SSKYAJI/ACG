+from __future__ import annotations
+
+import typing
+from urllib.parse import urlparse
+
+from starlette.datastructures import Headers, MutableHeaders
+from starlette.responses import PlainTextResponse, Response
+from starlette.types import ASGIApp, Message, Receive, Scope, Send
+
+
+class CORSMiddleware:
+    def __init__(
+        self,
+        app: ASGIApp,
+        allow_origins: typing.Sequence[str] = (),
+        allow_methods: typing.Sequence[str] = ("GET",),
+        allow_headers: typing.Sequence[str] = (),
+        allow_credentials: bool = False,
+        allow_origin_regex: str | None = None,
+        expose_headers: typing.Sequence[str] = (),
+        max_age: int = 600,
+    ) -> None:
+        self.app = app
+        self.allow_origins = list(allow_origins)
+        self.allow_methods = list(allow_methods)
+        self.allow_headers = list(allow_headers)
+        self.allow_credentials = allow_credentials
+        self.allow_origin_regex = allow_origin_regex
+        self.expose_headers = list(expose_headers)
+        self.max_age = max_age
+
+    async def __call__(self, scope: Scope, receive: Receive, send: Send) -> None:
+        if scope["type"] != "http":
+            await self.app(scope, receive, send)
+            return
+
+        method = scope["method"]
+        headers = Headers(scope=scope)
+        origin = headers.get("origin")
+
+        if origin is None:
+            # Not a CORS request
+            await self.app(scope, receive, send)
+            return
+
+        # Determine if origin is allowed
+        allowed = self.is_origin_allowed(origin)
+
+        if method == "OPTIONS" and "access-control-request-method" in headers:
+            # Preflight request
+            await self._preflight(scope, receive, send, origin, allowed)
+        else:
+            # Simple / actual request
+            await self._simple(scope, receive, send, origin, allowed)
+
+    def is_origin_allowed(self, origin: str) -> bool:
+        if "*" in self.allow_origins:
+            return True
+        if origin in self.allow_origins:
+            return True
+        if self.allow_origin_regex is not None:
+            import re
+            if re.match(self.allow_origin_regex, origin):
+                return True
+        return False
+
+    async def _preflight(
+        self, scope: Scope, receive: Receive, send: Send, origin: str, allowed: bool
+    ) -> None:
+        headers = MutableHeaders(scope=scope)
+        if allowed:
+            # For preflight with credentials, echo the origin
+            if self.allow_credentials and "*" in self.allow_origins:
+                headers["Access-Control-Allow-Origin"] = origin
+            else:
+                headers["Access-Control-Allow-Origin"] = "*" if "*" in self.allow_origins else origin
+            headers["Access-Control-Allow-Methods"] = ", ".join(self.allow_methods)
+            headers["Access-Control-Allow-Headers"] = ", ".join(self.allow_headers)
+            headers["Access-Control-Max-Age"] = str(self.max_age)
+            if self.allow_credentials:
+                headers["Access-Control-Allow-Credentials"] = "true"
+        else:
+            headers["Access-Control-Allow-Origin"] = ""
+        response = PlainTextResponse("", status_code=200, headers=headers)
+        await response(scope, receive, send)
+
+    async def _simple(
+        self, scope: Scope, receive: Receive, send: Send, origin: str, allowed: bool
+    ) -> None:
+        # Wrap the send to inject CORS headers
+        original_send = send
+
+        async def send_wrapper(message: Message) -> None:
+            if message["type"] == "http.response.start":
+                headers = MutableHeaders(scope=scope)
+                if allowed:
+                    # For simple requests with credentials and wildcard origin, echo the origin
+                    if self.allow_credentials and "*" in self.allow_origins:
+                        headers["Access-Control-Allow-Origin"] = origin
+                    else:
+                        headers["Access-Control-Allow-Origin"] = "*" if "*" in self.allow_origins else origin
+                    if self.allow_credentials:
+                        headers["Access-Control-Allow-Credentials"] = "true"
+                    if self.expose_headers:
+                        headers["Access-Control-Expose-Headers"] = ", ".join(self.expose_headers)
+                else:
+                    headers["Access-Control-Allow-Origin"] = ""
+            await original_send(message)
+
+        await self.app(scope, receive, send_wrapper)

+import pytest
+from starlette.applications import Starlette
+from starlette.middleware.cors import CORSMiddleware
+from starlette.responses import JSONResponse
+from starlette.routing import Route
+from starlette.testclient import TestClient
+
+
+def test_simple_request_with_credentials_and_wildcard_echoes_origin():
+    """
+    When allow_origins=['*'] and allow_credentials=True, a simple request
+    should echo the request Origin instead of '*'.
+    """
+    app = Starlette()
+    app.add_middleware(
+        CORSMiddleware,
+        allow_origins=["*"],
+        allow_credentials=True,
+        allow_methods=["GET"],
+    )
+
+    @app.route("/")
+    async def homepage(request):
+        return JSONResponse({"hello": "world"})
+
+    client = TestClient(app)
+    response = client.get("/", headers={"Origin": "https://example.com"})
+    assert response.status_code == 200
+    # The Access-Control-Allow-Origin should be the specific origin, not '*'
+    assert response.headers["access-control-allow-origin"] == "https://example.com"
+    assert response.headers.get("access-control-allow-credentials") == "true"
+
+
+def test_preflight_with_credentials_and_wildcard_echoes_origin():
+    """
+    Preflight requests should also echo the origin when credentials and wildcard are used.
+    """
+    app = Starlette()
+    app.add_middleware(
+        CORSMiddleware,
+        allow_origins=["*"],
+        allow_credentials=True,
+        allow_methods=["GET", "POST"],
+        allow_headers=["Content-Type"],
+    )
+
+    @app.route("/")
+    async def homepage(request):
+        return JSONResponse({"hello": "world"})
+
+    client = TestClient(app)
+    response = client.options(
+        "/",
+        headers={
+            "Origin": "https://example.com",
+            "Access-Control-Request-Method": "GET",
+        },
+    )
+    assert response.status_code == 200
+    assert response.headers["access-control-allow-origin"] == "https://example.com"
+    assert response.headers.get("access-control-allow-credentials") == "true"
+
+
+def test_simple_request_without_credentials_uses_wildcard():
+    """
+    When allow_origins=['*'] and allow_credentials=False, the origin should be '*'.
+    """
+    app = Starlette()
+    app.add_middleware(
+        CORSMiddleware,
+        allow_origins=["*"],
+        allow_credentials=False,
+    )
+
+    @app.route("/")
+    async def homepage(request):
+        return JSONResponse({"hello": "world"})
+
+    client = TestClient(app)
+    response = client.get("/", headers={"Origin": "https://example.com"})
+    assert response.status_code == 200
+    assert response.headers["access-control-allow-origin"] == "*"
+    assert "access-control-allow-credentials" not in response.headers
