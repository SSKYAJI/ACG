+from __future__ import annotations
+
+import typing
+from pathlib import Path
+
+from jinja2 import Environment, FileSystemLoader, select_autoescape
+from starlette.requests import Request
+from starlette.responses import HTMLResponse
+from starlette.types import ASGIApp, Receive, Scope, Send
+
+
+class _TemplateResponse(HTMLResponse):
+    def __init__(
+        self,
+        template: typing.Any,
+        context: dict,
+        status_code: int = 200,
+        headers: dict | None = None,
+        media_type: str | None = None,
+    ) -> None:
+        self.template = template
+        self.context = context
+        super().__init__(
+            content="",  # placeholder, will be rendered later
+            status_code=status_code,
+            headers=headers,
+            media_type=media_type,
+        )
+
+    async def __call__(self, scope: Scope, receive: Receive, send: Send) -> None:
+        request = Request(scope, receive)
+        context = self.context.copy()
+        context["request"] = request
+        content = self.template.render(context)
+        self.body = content.encode("utf-8")
+        await super().__call__(scope, receive, send)
+
+
+class Jinja2Templates:
+    def __init__(
+        self,
+        directory: str | Path | typing.Sequence[str | Path],
+        *,
+        env: Environment | None = None,
+        **env_options: typing.Any,
+    ) -> None:
+        self.env = env or self._create_environment(directory, **env_options)
+        self.env.globals.setdefault("url_for", None)
+
+    def _create_environment(
+        self,
+        directory: str | Path | typing.Sequence[str | Path],
+        **options: typing.Any,
+    ) -> Environment:
+        loader = FileSystemLoader(directory)
+        # Use select_autoescape by default, but allow caller to override
+        if "autoescape" not in options:
+            options["autoescape"] = select_autoescape(
+                enabled_extensions=("html", "htm", "xml"),
+                default_for_string=False,
+                default=False,
+            )
+        return Environment(loader=loader, **options)
+
+    def get_template(self, name: str) -> typing.Any:
+        return self.env.get_template(name)
+
+    def TemplateResponse(
+        self, name: str, context: dict, status_code: int = 200
+    ) -> _TemplateResponse:
+        template = self.get_template(name)
+        return _TemplateResponse(template, context, status_code=status_code)

+import pytest
+from starlette.templating import Jinja2Templates
+from starlette.testclient import TestClient
+from starlette.applications import Starlette
+from starlette.responses import HTMLResponse
+from starlette.routing import Route
+
+
+def test_autoescape_default():
+    """
+    Verify that by default Jinja2Templates enables autoescape for .html files.
+    """
+    import tempfile, os
+    from pathlib import Path
+
+    with tempfile.TemporaryDirectory() as tmpdir:
+        # Create a template with HTML content that would be escaped
+        template_path = Path(tmpdir) / "test.html"
+        template_path.write_text("<p>{{ content }}</p>")
+
+        templates = Jinja2Templates(directory=tmpdir)
+        app = Starlette(
+            routes=[
+                Route(
+                    "/",
+                    endpoint=lambda r: templates.TemplateResponse(
+                        "test.html", {"content": "<script>alert('xss')</script>"}
+                    ),
+                )
+            ]
+        )
+        client = TestClient(app)
+        response = client.get("/")
+        assert response.status_code == 200
+        # The script tag should be escaped
+        assert "&lt;script&gt;alert(&#39;xss&#39;)&lt;/script&gt;" in response.text
+
+
+def test_autoescape_disabled_when_env_provided():
+    """
+    When a caller-supplied environment is passed, its autoescape setting should be preserved.
+    """
+    import tempfile
+    from pathlib import Path
+    from jinja2 import Environment, FileSystemLoader

+                Route(
+                    "/",
+                    endpoint=lambda r: templates.TemplateResponse(
+                        "test.html", {"content": "<script>alert('xss')</script>"}
+                    ),
+                )
+            ]
+        )
+        client = TestClient(app)
+        response = client.get("/")
+        assert response.status_code == 200
+        # With autoescape disabled, the raw HTML should appear
+        assert "<script>alert('xss')</script>" in response.text
