+ class SessionMiddleware:
+     def __init__(
+         self,
+         app: ASGIApp,
+         secret_key: str,
+         session_cookie: str = "session",
+         max_age: int = 14 * 24 * 60 * 60,  # 14 days, in seconds
+         same_site: str = "lax",
+         https_only: bool = False,
+         domain: str = None,
+     ) -> None:
+         self.app = app
+         self.serializer = URLSafeTimedSerializer(
+             secret_key=secret_key,
+             salt="cookie-session",
+             serializer=json,
+             signer_kwargs={"key_derivation": "hmac"},
+         )
+         self.session_cookie = session_cookie
+         self.max_age = max_age
+         self.same_site = same_site
+         self.https_only = https_only
+         self.domain = domain
+ 
+     async def __call__(self, scope: Scope, receive: Receive, send: Send) -> None:
+         if scope["type"] not in ("http", "websocket"):
+             await self.app(scope, receive, send)
+             return
+ 
+         connection = HTTPConnection(scope)
+         initial_session_was_empty = True
+ 
+         if self.session_cookie in connection.cookies:
+             try:
+                 data = self.serializer.loads(
+                     connection.cookies[self.session_cookie],
+                     max_age=self.max_age,
+                 )
+                 session = Session(data)
+                 initial_session_was_empty = False
+             except BadSignature:
+                 session = Session()
+         else:
+             session = Session()
+ 
+         scope["session"] = session
+ 
+         async def send_wrapper(message: dict) -> None:
+             if message["type"] == "http.response.start":
+                 if session.modified:
+                     headers = MutableHeaders(raw=message["headers"])
+                     if session:
+                         cookie_value = self.serializer.dumps(dict(session))
+                         headers.append(
+                             "set-cookie",
+                             f"{self.session_cookie}={cookie_value}; "
+                             f"Max-Age={self.max_age}; "
+                             f"Path=/; "
+                             f"SameSite={self.same_site}; "
+                             f"{'Secure; ' if self.https_only else ''}"
+                             f"{f'Domain={self.domain}; ' if self.domain else ''}"
+                             f"HttpOnly",
+                         )
+                     elif not initial_session_was_empty:
+                         # Session was cleared, expire cookie
+                         headers.append(
+                             "set-cookie",
+                             f"{self.session_cookie}=; "
+                             f"Max-Age=0; "
+                             f"Path=/; "
+                             f"SameSite={self.same_site}; "
+                             f"{'Secure; ' if self.https_only else ''}"
+                             f"{f'Domain={self.domain}; ' if self.domain else ''}"
+                             f"HttpOnly",
+                         )
+             await send(message)
+ 
+         await self.app(scope, receive, send_wrapper)
