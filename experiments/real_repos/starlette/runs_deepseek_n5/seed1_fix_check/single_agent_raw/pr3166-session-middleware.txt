+import pytest
+from starlette.applications import Starlette
+from starlette.middleware.sessions import SessionMiddleware
+from starlette.responses import JSONResponse
+from starlette.routing import Route
+from starlette.testclient import TestClient
+
+
+def test_read_only_session_no_cookie():
+    app = Starlette()
+    app.add_middleware(SessionMiddleware, secret_key="test")
+
+    @app.route("/read")
+    async def read(request):
+        session = request.session
+        # just access, no mutation
+        return JSONResponse({"count": session.get("count", 0)})
+
+    client = TestClient(app)
+    response = client.get("/read")
+    assert response.status_code == 200
+    # No Set-Cookie header because session was not modified
+    assert "set-cookie" not in response.headers
+
+
+def test_mutate_session_sets_cookie():
+    app = Starlette()
+    app.add_middleware(SessionMiddleware, secret_key="test")
+
+    @app.route("/set")
+    async def set_session(request):
+        request.session["key"] = "value"
+        return JSONResponse({"ok": True})
+
+    client = TestClient(app)
+    response = client.get("/set")
+    assert response.status_code == 200
+    assert "set-cookie" in response.headers
+
+
+def test_clear_session_deletes_cookie():
+    app = Starlette()
+    app.add_middleware(SessionMiddleware, secret_key="test")
+
+    @app.route("/clear")
+    async def clear_session(request):
+        request.session.clear()
+        return JSONResponse({"ok": True})
+
+    client = TestClient(app)
+    # First set a session
+    client.get("/set")
+    # Then clear
+    response = client.get("/clear")
+    assert response.status_code == 200
+    assert "set-cookie" in response.headers
+    # The cookie value should be empty or expired (deletion)
+    # We can check that the header contains 'session=;' or similar
+    assert "session=;" in response.headers["set-cookie"] or "Max-Age=0" in response.headers["set-cookie"]
