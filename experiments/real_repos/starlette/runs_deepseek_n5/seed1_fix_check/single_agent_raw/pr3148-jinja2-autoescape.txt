+import pytest
+from starlette.templating import Jinja2Templates

+def test_autoescape_html():
+    templates = Jinja2Templates(directory="tests/templates")
+    app = Starlette(
+        routes=[
+            Route("/", endpoint=lambda r: templates.TemplateResponse("test.html", {"request": r, "name": "<b>bold</b>"}))
+        ]
+    )
+    client = TestClient(app)
+    response = client.get("/")
+    assert response.status_code == 200
+    assert "&lt;b&gt;bold&lt;/b&gt;" in response.text
+
