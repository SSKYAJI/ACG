+class SessionDict(dict):
+    """A dict subclass that tracks whether it has been modified."""
+    def __init__(self, *args, **kwargs):
+        super().__init__(*args, **kwargs)
+        self._modified = False
+
+    def __setitem__(self, key, value):
+        super().__setitem__(key, value)
+        self._modified = True
+
+    def __delitem__(self, key):
+        super().__delitem__(key)
+        self._modified = True
+
+    def pop(self, *args, **kwargs):
+        result = super().pop(*args, **kwargs)
+        self._modified = True
+        return result
+
+    def popitem(self):
+        result = super().popitem()
+        self._modified = True
+        return result
+
+    def clear(self):
+        super().clear()
+        self._modified = True
+
+    def update(self, *args, **kwargs):
+        super().update(*args, **kwargs)
+        self._modified = True
+
+    def setdefault(self, key, default=None):
+        if key not in self:
+            super().__setitem__(key, default)
+            self._modified = True
+        return self[key]

+def test_session_clear_clears_cookie():
+    app = Starlette()
+    app.add_middleware(SessionMiddleware, secret_key="test")
+
+    @app.route("/set")
+    async def set_session(request):
+        request.session["key"] = "value"
+        return PlainTextResponse("OK")
+
+    @app.route("/clear")
+    async def clear_session(request):
+        request.session.clear()
+        return PlainTextResponse("OK")
+
+    client = TestClient(app)
+    # First set a session
+    response = client.get("/set")
+    cookie = response.headers["set-cookie"]
+    # Then clear it
+    response = client.get("/clear", cookies={"session": cookie.split(";")[0].split("=")[1]})
+    assert response.status_code == 200
+    set_cookie = response.headers.get("set-cookie")
+    assert set_cookie is not None
+    # The cookie should have max-age=0 or be expired
+    assert "max-age=0" in set_cookie or "expires=" in set_cookie
