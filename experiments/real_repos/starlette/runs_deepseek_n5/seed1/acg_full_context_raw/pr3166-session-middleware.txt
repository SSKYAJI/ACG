     def __init__(

         self.session_cookie = session_cookie
         self.max_age = max_age
         self.same_site = same_site
         self.https_only = https_only
         self.domain = domain
         self.path = path
+        self._mutated_sessions: Set[str] = set()  # track session keys that were mutated

     async def __call__(self, scope: Scope, receive: Receive, send: Send) -> None:
         if scope["type"] != "http":  # pragma: no cover
             await self.app(scope, receive, send)
             return

         connection = HTTPConnection(scope)
         initial_session_serialized = connection.cookies.get(self.session_cookie)

         if initial_session_serialized:
             try:
                 data = self.signer.unsign(initial_session_serialized)
                 session = MutableSession(data)
             except (BadSignature, SignatureExpired):
                 session = MutableSession()
         else:
             session = MutableSession()

         scope["session"] = session

         # Wrap send to intercept response and conditionally set cookie
         async def send_wrapper(message: Message) -> None:
             if message["type"] == "http.response.start":
                 if session.accessed and not session.modified:
                     # Session was read but not mutated; do not set cookie
                     pass
                 elif session.modified:
                     # Session was mutated; set cookie
                     if session:
                         # If session is not empty, set cookie
                         cookie_value = self.signer.sign(session.data)
                         headers = MutableHeaders(scope=message)
                         headers.append(
                             "Set-Cookie",
                             "%s=%s; Max-Age=%s; Path=%s; SameSite=%s; %s"
                             % (
                                 self.session_cookie,
                                 cookie_value,
                                 self.max_age,
                                 self.path,
                                 self.same_site,
                                 "Secure; " if self.https_only else "",
                             )
                             + (f" Domain={self.domain};" if self.domain else "")
                         )
                     else:
                         # Session was mutated to empty; clear cookie
                         headers = MutableHeaders(scope=message)
                         headers.append(
                             "Set-Cookie",
                             "%s=; Max-Age=0; Path=%s; SameSite=%s; %s"
                             % (
                                 self.session_cookie,
                                 self.path,
                                 self.same_site,
                                 "Secure; " if self.https_only else "",
                             )
                             + (f" Domain={self.domain};" if self.domain else "")
                         )
                 # If session was not accessed at all (e.g., never read), do nothing
             await send(message)

         await self.app(scope, receive, send_wrapper)
