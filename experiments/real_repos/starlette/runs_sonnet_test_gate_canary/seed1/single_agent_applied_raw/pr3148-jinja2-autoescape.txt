+class _TemplateResponse:
+    def __init__(self, template, context, status_code=200, headers=None, media_type=None, background=None):
+        self.template = template
+        self.context = context
+        self.status_code = status_code
+        self.headers = headers or {}
+        self.media_type = media_type or "text/html"
+        self.background = background
+        self.body = self._render()
+
+    def _render(self):
+        return self.template.render(self.context).encode("utf-8")
+
+
+class Jinja2Templates:
+    """
+    templates = Jinja2Templates("templates")
+
+    return templates.TemplateResponse("index.html", {"request": request})
+    """

+    def __init__(
+        self,
+        directory: typing.Union[str, None] = None,
+        *,
+        env: typing.Union["Environment", None] = None,
+    ) -> None:
+        assert jinja2 is not None, "jinja2 must be installed to use Jinja2Templates"
+        assert (
+            directory is not None or env is not None
+        ), "either 'directory' or 'env' arguments must be passed"
+        if directory is not None:
+            self.env = Environment(
+                loader=FileSystemLoader(directory),
+                autoescape=select_autoescape(),
+            )
+        else:
+            self.env = env  # type: ignore

+    def get_template(self, name: str) -> "jinja2.Template":
+        return self.env.get_template(name)

+    def TemplateResponse(
+        self,
+        name: str,
+        context: dict,
+        status_code: int = 200,
+        headers: typing.Optional[typing.Dict[str, str]] = None,
+        media_type: typing.Optional[str] = None,
+        background: typing.Optional[typing.Any] = None,
+    ) -> _TemplateResponse:
+        template = self.get_template(name)
+        return _TemplateResponse(
+            template,
+            context,
+            status_code=status_code,
+            headers=headers,
+            media_type=media_type,
+            background=background,
+        )

+class TestJinja2TemplatesAutoescape:
+    def test_autoescape_enabled_for_html_by_default(self, template_dir):
+        """HTML templates should have autoescape enabled by default."""
+        write_template(template_dir, "index.html", "<p>{{ value }}</p>")
+        templates = Jinja2Templates(directory=str(template_dir))
+        template = templates.get_template("index.html")
+        result = template.render({"value": "<script>alert('xss')</script>"})
+        assert "<script>" not in result
+        assert "&lt;script&gt;" in result

+    def test_autoescape_enabled_for_xml_by_default(self, template_dir):
+        """XML templates should have autoescape enabled by default."""
+        write_template(template_dir, "feed.xml", "<item>{{ value }}</item>")
+        templates = Jinja2Templates(directory=str(template_dir))
+        template = templates.get_template("feed.xml")
+        result = template.render({"value": "<bad>&</bad>"})
+        assert "<bad>" not in result
+        assert "&lt;bad&gt;" in result

+    def test_autoescape_not_enabled_for_txt_by_default(self, template_dir):
+        """Plain text templates should NOT have autoescape enabled by default."""
+        write_template(template_dir, "email.txt", "Hello {{ value }}")
+        templates = Jinja2Templates(directory=str(template_dir))
+        template = templates.get_template("email.txt")
+        result = template.render({"value": "<b>world</b>"})
+        # No escaping for .txt files
+        assert "<b>world</b>" in result

+    def test_caller_supplied_env_is_preserved(self, template_dir):

+        write_template(template_dir, "page.html", "<p>{{ value }}</p>")
+        # Create an env with autoescape explicitly disabled
+        env = Environment(
+            loader=FileSystemLoader(str(template_dir)),

+        )
+        templates = Jinja2Templates(env=env)

+        result = template.render({"value": "<script>xss</script>"})
+        # autoescape is disabled in caller-supplied env, so raw HTML passes through
