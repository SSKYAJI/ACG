"""Regression tests for Enum field by-name lookup (pr2902)."""
import enum
import pytest
from marshmallow.fields import Enum as EnumField
from marshmallow import ValidationError


class Color(enum.Enum):
    RED = 1
    GREEN = 2
    BLUE = 3

    def describe(self):
        return f"I am {self.name}"

    @property
    def hex_value(self):
        return f"#{self.value:06x}"

    @classmethod
    def from_string(cls, s):
        return cls[s.upper()]


class Status(enum.Enum):
    ACTIVE = "active"
    INACTIVE = "inactive"
    PENDING = "pending"


class TestEnumByNameLookup:
    """Tests for Enum field by-name lookup."""

    def test_valid_member_by_name(self):
        """Enum field should return enum member when looking up by valid name."""
        field = EnumField(Color, by_value=False)
        result = field._deserialize("RED", "field", {})
        assert result == Color.RED

    def test_valid_member_by_name_green(self):
        """Enum field should return GREEN member when looking up by name."""
        field = EnumField(Color, by_value=False)
        result = field._deserialize("GREEN", "field", {})
        assert result == Color.GREEN

    def test_valid_member_by_name_blue(self):
        """Enum field should return BLUE member when looking up by name."""
        field = EnumField(Color, by_value=False)
        result = field._deserialize("BLUE", "field", {})
        assert result == Color.BLUE

    def test_method_name_raises_validation_error(self):
        """Enum field should raise ValidationError when looking up a method name."""
        field = EnumField(Color, by_value=False)
        # 'describe' is a method on Color, not an enum member
        with pytest.raises(ValidationError):
            field._deserialize("describe", "field", {})

    def test_property_name_raises_validation_error(self):
        """Enum field should raise ValidationError when looking up a property name."""
        field = EnumField(Color, by_value=False)
        # 'hex_value' is a property on Color, not an enum member
        with pytest.raises(ValidationError):
            field._deserialize("hex_value", "field", {})

    def test_classmethod_name_raises_validation_error(self):
        """Enum field should raise ValidationError when looking up a classmethod name."""
        field = EnumField(Color, by_value=False)
        # 'from_string' is a classmethod on Color, not an enum member
        with pytest.raises(ValidationError):
            field._deserialize("from_string", "field", {})

    def test_nonexistent_name_raises_validation_error(self):
        """Enum field should raise ValidationError for nonexistent names."""
        field = EnumField(Color, by_value=False)
        with pytest.raises(ValidationError):
            field._deserialize("YELLOW", "field", {})

    def test_load_by_name_valid_member(self):
        """Enum field with load_by_name=True should return enum member."""
        field = EnumField(Color, load_by_name=True)
        result = field._deserialize("RED", "field", {})
        assert result == Color.RED

    def test_load_by_name_method_raises_validation_error(self):
        """Enum field with load_by_name=True should raise ValidationError for method names."""
        field = EnumField(Color, load_by_name=True)
        with pytest.raises(ValidationError):
            field._deserialize("describe", "field", {})

    def test_load_by_name_property_raises_validation_error(self):
        """Enum field with load_by_name=True should raise ValidationError for property names."""
        field = EnumField(Color, load_by_name=True)
        with pytest.raises(ValidationError):
            field._deserialize("hex_value", "field", {})

    def test_by_value_still_works(self):
        """Enum field with by_value=True should still work correctly."""
        field = EnumField(Color, by_value=True)
        result = field._deserialize(1, "field", {})
        assert result == Color.RED

    def test_by_value_invalid_raises_validation_error(self):
        """Enum field with by_value=True should raise ValidationError for invalid values."""
        field = EnumField(Color, by_value=True)
        with pytest.raises(ValidationError):
            field._deserialize(99, "field", {})

    def test_member_map_only_contains_members(self):
        """_member_map_ should only contain actual enum members."""
        assert "RED" in Color._member_map_
        assert "GREEN" in Color._member_map_
        assert "BLUE" in Color._member_map_
        assert "describe" not in Color._member_map_
        assert "hex_value" not in Color._member_map_
        assert "from_string" not in Color._member_map_

    def test_string_enum_by_name(self):
        """Enum field should work with string-valued enums by name."""
        field = EnumField(Status, by_value=False)
        result = field._deserialize("ACTIVE", "field", {})
        assert result == Status.ACTIVE

    def test_string_enum_by_name_invalid_raises(self):
        """Enum field should raise ValidationError for invalid names in string enum."""
        field = EnumField(Status, by_value=False)
        with pytest.raises(ValidationError):
            field._deserialize("active", "field", {})  # lowercase, not a member name
