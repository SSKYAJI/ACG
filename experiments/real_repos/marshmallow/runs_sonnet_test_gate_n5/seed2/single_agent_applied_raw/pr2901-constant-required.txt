"""Regression tests for Constant field with required=True (pr2901)."""
import pytest
from marshmallow.fields import Constant


class TestConstantRequired:
    """Tests for Constant field with required=True."""

    def test_constant_default_not_required(self):
        """Constant field should work without required=True by default."""
        field = Constant("hello")
        assert field.required is False
        assert field.constant == "hello"

    def test_constant_with_required_true_does_not_raise(self):
        """Constant field should accept required=True without raising ValueError."""
        # This should not raise a ValueError
        field = Constant("hello", required=True)
        assert field.required is True
        assert field.constant == "hello"

    def test_constant_with_required_false_does_not_raise(self):
        """Constant field should accept required=False without raising ValueError."""
        field = Constant("hello", required=False)
        assert field.required is False
        assert field.constant == "hello"

    def test_constant_required_true_serializes_constant(self):
        """Constant field with required=True should serialize to constant value."""
        field = Constant(42, required=True)
        result = field._serialize(None, "field", {})
        assert result == 42

    def test_constant_required_true_deserializes_constant(self):
        """Constant field with required=True should deserialize to constant value."""
        field = Constant(42, required=True)
        result = field._deserialize("anything", "field", {})
        assert result == 42

    def test_constant_required_true_with_string(self):
        """Constant field with required=True should work with string constants."""
        field = Constant("fixed_value", required=True)
        assert field.required is True
        assert field.constant == "fixed_value"
        assert field._serialize(None, "f", {}) == "fixed_value"
        assert field._deserialize("x", "f", {}) == "fixed_value"

    def test_constant_required_true_with_none_constant(self):
        """Constant field with required=True should work with None as constant."""
        field = Constant(None, required=True)
        assert field.required is True
        assert field.constant is None

    def test_constant_required_true_with_dict_constant(self):
        """Constant field with required=True should work with dict constants."""
        const_val = {"key": "value"}
        field = Constant(const_val, required=True)
        assert field.required is True
        assert field.constant == const_val

    def test_constant_required_true_with_list_constant(self):
        """Constant field with required=True should work with list constants."""
        const_val = [1, 2, 3]
        field = Constant(const_val, required=True)
        assert field.required is True
        assert field.constant == const_val

    def test_constant_load_default_set_to_constant(self):
        """Constant field should set load_default to the constant value."""
        field = Constant("test", required=True)
        assert field.load_default == "test"

    def test_constant_dump_default_set_to_constant(self):
        """Constant field should set dump_default to the constant value."""
        field = Constant("test", required=True)
        assert field.dump_default == "test"
