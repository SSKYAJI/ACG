+ class Field:
+     default_error_messages = {
+         "required": "Missing data for required field.",
+         "null": "Field may not be null.",
+         "validator_failed": "Invalid value.",
+     }
+
+     def __init__(self, required=False, default=missing_, missing=missing_,
+                  allow_none=False, load_only=False, dump_only=False,
+                  validate=None, metadata=None, **kwargs):
+         self.required = required
+         self.default = default
+         self.missing = missing
+         self.allow_none = allow_none
+         self.load_only = load_only
+         self.dump_only = dump_only
+         if validate is None:
+             self.validators = []
+         elif callable(validate):
+             self.validators = [validate]
+         else:
+             self.validators = list(validate)
+         self.metadata = metadata or {}
+         self.name = None
+         self.parent = None
+
+     def serialize(self, attr, obj, accessor=None):
+         value = self.get_value(obj, attr, accessor=accessor)
+         if value is missing_:
+             default = self.default
+             value = default() if callable(default) else default
+         if value is missing_:
+             return None
+         return self._serialize(value, attr, obj)
+
+     def deserialize(self, value, attr=None, data=None):
+         if value is missing_:
+             _miss = self.missing
+             _miss = _miss() if callable(_miss) else _miss
+             if _miss is missing_:
+                 if self.required:
+                     self.fail("required")
+             return _miss
+         if value is None and not self.allow_none:
+             self.fail("null")
+         output = self._deserialize(value, attr, data)
+         self._validate(output)
+         return output
+
+     def _serialize(self, value, attr, obj):
+         return value
+
+     def _deserialize(self, value, attr, data):
+         return value
+
+     def _validate(self, value):
+         errors = []
+         for validator in self.validators:
+             try:
+                 result = validator(value)
+                 if result is False:
+                     self.fail("validator_failed")
+             except ValidationError as e:
+                 if isinstance(e.messages, list):
+                     errors.extend(e.messages)
+                 else:
+                     errors.append(e.messages)
+         if errors:
+             raise ValidationError(errors)
+
+     def get_value(self, obj, attr, accessor=None, default=missing_):
+         if accessor is not None:
+             return accessor(obj, attr, default)
+         if isinstance(obj, dict):
+             return obj.get(attr, default)
+         return getattr(obj, attr, default)
+
+     def fail(self, key, **kwargs):
+         msg = self.default_error_messages.get(key, key)
+         raise ValidationError(msg)

+ class Constant(Field):
+     """A field that always serializes/deserializes to a constant value.
+
+     The ``required`` parameter is accepted but has no practical effect since
+     the constant value is always provided.
+     """
+
+     def __init__(self, constant, **kwargs):
+         # Constant always provides a value, so required=True should not raise.
+         # We set the missing value to the constant so deserialization always
+         # returns the constant even when the field is absent from input.
+         super().__init__(**kwargs)
+         self.constant = constant
+         # Override missing so that a missing input still yields the constant.
+         self.missing = constant
+         # Override default so serialization always yields the constant.
+         self.default = constant
+
+     def _serialize(self, value, attr, obj):
+         return self.constant
+
+     def _deserialize(self, value, attr, data):
+         return self.constant

+ class SchemaMeta(type):
+     def __new__(mcs, name, bases, attrs):
+         fields = {}
+         for base in bases:
+             if hasattr(base, '_declared_fields'):
+                 fields.update(base._declared_fields)
+         for key, value in list(attrs.items()):
+             if isinstance(value, Field):
+                 fields[key] = value
+                 value.name = key
+         attrs['_declared_fields'] = fields
+         return super().__new__(mcs, name, bases, attrs)
+
+
+ class Schema(metaclass=SchemaMeta):
+     def __init__(self, many=False, context=None):
+         self.many = many
+         self.context = context or {}
+         self.fields = {}
+         for name, field_obj in self._declared_fields.items():
+             field_obj.name = name
+             field_obj.parent = self
+             self.fields[name] = field_obj
+
+     def dump(self, obj):
+         if self.many:
+             return [self._serialize(o) for o in obj]
+         return self._serialize(obj)
+
+     def load(self, data):
+         if self.many:
+             return [self._deserialize(d) for d in data]
+         return self._deserialize(data)
+
+     def _serialize(self, obj):
+         result = {}
+         for name, field_obj in self.fields.items():
+             value = field_obj.serialize(name, obj)
+             result[name] = value
+         return result
+
+     def _deserialize(self, data):
+         result = {}
+         errors = {}
+         for name, field_obj in self.fields.items():
+             raw_value = data.get(name, missing_) if isinstance(data, dict) else missing_
+             try:
+                 value = field_obj.deserialize(raw_value, name, data)
+                 if value is not missing_:
+                     result[name] = value
+             except ValidationError as e:
+                 errors[name] = e.messages
+         if errors:
+             raise ValidationError(errors)
+         return result
