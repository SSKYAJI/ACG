+ class Constant(Field):
+     """A field that always serializes/deserializes to a constant value.
+
+     Args:
+         constant: The constant value to use.
+         **kwargs: The same keyword arguments that :class:`Field` receives.
+             Note: ``required=True`` is accepted and does not raise an error,
+             since the constant value is always provided.
+     """
+
+     def __init__(self, constant, **kwargs):
+         # A Constant field always provides its value, so required=True is
+         # valid and should not raise a ValueError.
+         # We set load_default to the constant value so that even if the field
+         # is missing from input, the constant is returned.
+         if "load_default" not in kwargs and "missing" not in kwargs:
+             kwargs["load_default"] = constant
+         super().__init__(**kwargs)
+         self.constant = constant
+
+     def _serialize(self, value, attr, obj, **kwargs):
+         return self.constant
+
+     def _deserialize(self, value, attr, data, **kwargs):
+         return self.constant
+
+     def deserialize(self, value, attr=None, data=None):
+         # Always return the constant value regardless of input.
+         # This means required=True is satisfied because we always have a value.
+         return self.constant

+ class SchemaMeta(type):
+     def __new__(mcs, name, bases, attrs):
+         fields = {}
+         # Collect fields from base classes
+         for base in reversed(bases):
+             if hasattr(base, "_declared_fields"):
+                 fields.update(base._declared_fields)
+         # Collect fields from current class
+         for key, value in list(attrs.items()):
+             if isinstance(value, Field):
+                 fields[key] = value
+         attrs["_declared_fields"] = fields
+         return super().__new__(mcs, name, bases, attrs)
+
+
+ class Schema(metaclass=SchemaMeta):
+     def __init__(self, *, only=None, exclude=(), many=False, context=None,
+                  load_only=(), dump_only=(), partial=False, unknown=None):
+         self.only = only
+         self.exclude = set(exclude)
+         self.many = many
+         self.context = context or {}
+         self.load_only = set(load_only)
+         self.dump_only = set(dump_only)
+         self.partial = partial
+         self.unknown = unknown
+
+         # Bind fields
+         self.fields = {}
+         for name, field_obj in self._declared_fields.items():
+             if self.only and name not in self.only:
+                 continue
+             if name in self.exclude:
+                 continue
+             field_obj.parent = self
+             field_obj.name = name
+             self.fields[name] = field_obj
+
+     def dump(self, obj, *, many=None):
+         many = self.many if many is None else many
+         if many:
+             return [self._serialize(item) for item in obj]
+         return self._serialize(obj)
+
+     def _serialize(self, obj):
+         result = {}
+         for name, field_obj in self.fields.items():
+             if field_obj.load_only:
+                 continue
+             value = field_obj.serialize(name, obj)
+             result[name] = value
+         return result
+
+     def load(self, data, *, many=None, partial=None):
+         many = self.many if many is None else many
+         if many:
+             return [self._deserialize(item) for item in data]
+         return self._deserialize(data)
+
+     def _deserialize(self, data):
+         result = {}
+         errors = {}
+         for name, field_obj in self.fields.items():
+             if field_obj.dump_only:
+                 continue
+             key = field_obj.data_key if field_obj.data_key else name
+             raw_value = data.get(key, missing_)
+             try:
+                 value = field_obj.deserialize(raw_value, key, data)
+                 result[name] = value
+             except ValidationError as e:
+                 errors[name] = e.messages
+         if errors:
+             raise ValidationError(errors)
+         return result
