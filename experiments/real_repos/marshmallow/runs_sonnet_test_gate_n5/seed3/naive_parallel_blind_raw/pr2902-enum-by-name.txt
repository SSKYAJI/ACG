+ from marshmallow import fields
+ from marshmallow.exceptions import ValidationError
+ import enum
+

+ class EnumField(fields.Field):
+     """A marshmallow field for serializing/deserializing Python enums."""
+
+     default_error_messages = {
+         "by_name": "Must be one of: {choices}.",
+         "by_value": "Must be one of: {choices}.",
+     }
+
+     def __init__(self, enum_type, by_value=False, *args, **kwargs):
+         self.enum_type = enum_type
+         self.by_value = by_value
+         super().__init__(*args, **kwargs)
+
+     def _serialize(self, value, attr, obj, **kwargs):
+         if value is None:
+             return None
+         if self.by_value:
+             return value.value
+         return value.name
+
+     def _deserialize(self, value, attr, data, **kwargs):
+         if self.by_value:
+             return self._deserialize_by_value(value, attr, data)
+         return self._deserialize_by_name(value, attr, data)
+
+     def _deserialize_by_name(self, value, attr, data, **kwargs):
+         try:
+             # Only look up actual enum members, not arbitrary class attributes
+             member = self.enum_type[value]
+         except KeyError:
+             choices = [m.name for m in self.enum_type]
+             raise self.make_error("by_name", choices=", ".join(choices))
+         return member
+
+     def _deserialize_by_value(self, value, attr, data, **kwargs):
+         try:
+             return self.enum_type(value)
+         except ValueError:
+             choices = [str(m.value) for m in self.enum_type]
+             raise self.make_error("by_value", choices=", ".join(choices))
