+class Email(Validator):
+    """Validate an email address.
+
+    :param error: Error message to display when validation fails.
+    """
+
+    USER_REGEX = re.compile(
+        r"(^[-!#$%&'*+/=?^`{}|~\w]+(\.[-!#$%&'*+/=?^`{}|~\w]+)*\Z"  # dot-atom
+        r'|^"([\001-\010\013\014\016-\037!#-\[\]-\177]'
+        r'|\\[\001-\011\013\014\016-\177])*"\Z)',  # quoted-string
+        re.IGNORECASE | re.UNICODE,
+    )
+
+    DOMAIN_REGEX = re.compile(
+        # domain
+        r"(?:[A-Z0-9](?:[A-Z0-9-]{0,61}[A-Z0-9])?\.)+(?:[A-Z]{2,6}|[A-Z0-9-]{2,})\Z"
+        # literal form, ipv4 address (SMTP 4.1.3)
+        r"|^\[(\d{1,3}\.){3}\d{1,3}\]\Z",
+        re.IGNORECASE,
+    )
+
+    DOMAIN_WHITELIST = ("localhost",)
+
+    default_message = "Not a valid email address."
+
+    def __init__(self, *, error: str = None):
+        self.error = error or self.default_message
+
+    def __call__(self, value) -> None:
+        message = self.error
+
+        if not value or "@" not in value:
+            raise ValidationError(message)
+
+        user_part, domain_part = value.rsplit("@", 1)
+
+        if not self.USER_REGEX.match(user_part):
+            raise ValidationError(message)
+
+        if domain_part not in self.DOMAIN_WHITELIST:
+            # Validate each label of the domain BEFORE IDNA encoding
+            # to catch labels starting or ending with hyphens (invalid per RFC 5321)
+            for label in domain_part.split("."):
+                if label.startswith("-") or label.endswith("-"):
+                    raise ValidationError(message)
+
+            # Try to encode the domain as IDNA (handles internationalized domains)
+            try:
+                domain_part = domain_part.encode("idna").decode("ascii")
+            except (UnicodeError, UnicodeDecodeError):
+                raise ValidationError(message)
+
+            if not self.DOMAIN_REGEX.match(domain_part):
+                raise ValidationError(message)
