+ import enum

+ import marshmallow.fields as fields
+ from marshmallow import ValidationError

+ class MyEnum(enum.Enum):
+     RED = 1
+     GREEN = 2
+     BLUE = 3
+ 
+     def my_method(self):
+         return "method"
+ 
+     @property
+     def my_property(self):
+         return "property"
+ 
+ 
+ class TestEnumByNameLookup:
+     def setup_method(self):
+         self.field = fields.Enum(MyEnum, by_value=False)
+ 
+     def test_valid_member_by_name(self):
+         assert self.field.deserialize("RED") == MyEnum.RED
+         assert self.field.deserialize("GREEN") == MyEnum.GREEN
+         assert self.field.deserialize("BLUE") == MyEnum.BLUE
+
+     def test_invalid_name_raises_validation_error(self):
+         with pytest.raises(ValidationError):
+             self.field.deserialize("YELLOW")
+ 
+     def test_method_name_raises_validation_error(self):
+         """Non-member attributes like methods should not be returned."""
+         with pytest.raises(ValidationError):
+             self.field.deserialize("my_method")
+ 
+     def test_property_name_raises_validation_error(self):
+         """Non-member attributes like properties should not be returned."""
+         with pytest.raises(ValidationError):
+             self.field.deserialize("my_property")
+ 
+     def test_dunder_name_raises_validation_error(self):
+         """Dunder attributes should not be returned as enum members."""
+         with pytest.raises(ValidationError):
+             self.field.deserialize("__class__")
+ 
+     def test_serialize_member_by_name(self):
+         assert self.field._serialize(MyEnum.RED, None, None) == "RED"
