+ """Regression tests for Enum field by-name lookup."""
+ import enum

+ import marshmallow.fields as fields

+ 
+ class Color(enum.Enum):
+     RED = 1
+     GREEN = 2
+     BLUE = 3
+ 
+     def describe(self):
+         return f"I am {self.name}"
+ 
+     @property
+     def label(self):
+         return self.name.lower()
+ 
+     @classmethod
+     def default(cls):
+         return cls.RED
+ 
+ 
+ class TestEnumByNameLookup:
+     def setup_method(self):
+         self.field = fields.Enum(Color)
+ 
+     def test_valid_member_by_name(self):

+     def test_invalid_name_raises_validation_error(self):
