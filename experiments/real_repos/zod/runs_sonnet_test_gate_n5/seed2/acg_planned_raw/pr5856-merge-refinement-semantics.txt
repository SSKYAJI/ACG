 export function mergeDefs(...defs: Record<string, any>[]): any {
   const mergedDescriptors: Record<string, PropertyDescriptor> = {};
 
   for (const def of defs) {
     const descriptors = Object.getOwnPropertyDescriptors(def);
     Object.assign(mergedDescriptors, descriptors);
   }
 
   return Object.defineProperties({}, mergedDescriptors);
 }
+
+/**
+ * Unwraps a ZodType from any wrapping ZodEffects layers, collecting the effect defs along the way.
+ * Returns the innermost non-effects schema and the list of effect definitions in outer-to-inner order.
+ */
+export function unwrapEffects(schema: schemas.$ZodType): {
+  inner: schemas.$ZodType;
+  effects: any[];
+} {
+  const effects: any[] = [];
+  let current: schemas.$ZodType = schema;
+  while ((current._zod.def as any).type === "effects") {
+    effects.push(current._zod.def);
+    current = (current._zod.def as any).schema;
+  }
+  return { inner: current, effects };
+}
+
+/**
+ * Merges two ZodObject schemas (possibly wrapped in ZodEffects) while preserving
+ * refinements from both. The shape/catchall of the incoming object overrides the base.
+ * Refinements from the base are applied first, then refinements from the incoming object.
+ */
+export function mergeObjectSchemas(
+  base: schemas.$ZodType,
+  incoming: schemas.$ZodType
+): schemas.$ZodType {
+  const { inner: baseInner, effects: baseEffects } = unwrapEffects(base);
+  const { inner: incomingInner, effects: incomingEffects } = unwrapEffects(incoming);
+
+  // Merge the raw object defs (shape, catchall, etc.)
+  const mergedDef = mergeDefs(baseInner._zod.def, incomingInner._zod.def);
+
+  // Reconstruct the merged object
+  let result: schemas.$ZodType = new baseInner._zod.constr(mergedDef);
+
+  // Re-apply base effects first, then incoming effects
+  const allEffects = [...baseEffects, ...incomingEffects];
+  for (const effectDef of allEffects) {
+    const newDef = { ...effectDef, schema: result };
+    result = new (result._zod.constr as any)(newDef);
+    // We need to use the effects constructor, not the object constructor
+    // The effect def carries its own constructor info via the type field
+  }
+
+  return result;
+}
