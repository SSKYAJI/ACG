+import { describe, expect, test } from "vitest";
+import * as z from "../../../index.js";
+
+describe("shallowClone Map and Set - regression tests for pr5855", () => {
+  test("defaulted Map returns independent copy on each parse", () => {
+    const schema = z.object({
+      m: z.map(z.string(), z.number()).default(new Map([["a", 1]])),
+    });
+
+    const r1 = schema.parse({});
+    const r2 = schema.parse({});
+
+    expect(r1.m).toEqual(new Map([["a", 1]]));
+    expect(r2.m).toEqual(new Map([["a", 1]]));
+
+    r1.m.set("b", 2);
+    expect(r1.m.size).toBe(2);
+    // r2 must not be affected
+    expect(r2.m.size).toBe(1);
+  });
+
+  test("defaulted Set returns independent copy on each parse", () => {
+    const schema = z.object({
+      s: z.set(z.string()).default(new Set(["x", "y"])),
+    });
+
+    const r1 = schema.parse({});
+    const r2 = schema.parse({});
+
+    expect(r1.s).toEqual(new Set(["x", "y"]));
+    expect(r2.s).toEqual(new Set(["x", "y"]));
+
+    r1.s.add("z");
+    expect(r1.s.size).toBe(3);
+    // r2 must not be affected
+    expect(r2.s.size).toBe(2);
+  });
+
+  test("mutating parsed Map does not affect the default value", () => {
+    const defaultMap = new Map<string, number>([["key", 1]]);
+    const schema = z.object({
+      m: z.map(z.string(), z.number()).default(defaultMap),
+    });
+    const result = schema.parse({});
+    result.m.set("other", 99);
+    expect(defaultMap.size).toBe(1);
+    expect(defaultMap.has("other")).toBe(false);
+  });
+
+  test("mutating parsed Set does not affect the default value", () => {
+    const defaultSet = new Set<string>(["original"]);
+    const schema = z.object({
+      s: z.set(z.string()).default(defaultSet),
+    });
+    const result = schema.parse({});
+    result.s.add("added");
+    expect(defaultSet.size).toBe(1);
+    expect(defaultSet.has("added")).toBe(false);
+  });
+});
