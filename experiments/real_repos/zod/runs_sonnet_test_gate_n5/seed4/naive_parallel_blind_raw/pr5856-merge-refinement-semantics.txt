+// Helper to merge ZodObject refinements (checks) from a source object into a target
+export function mergeObjectChecks<T extends { _def: { checks?: unknown[] } }>(
+  target: T,
+  source: { _def: { checks?: unknown[] } }
+): T {
+  const sourceChecks = source._def.checks;
+  if (!sourceChecks || sourceChecks.length === 0) return target;
+  const targetDef = (target as any)._def;
+  targetDef.checks = [...(targetDef.checks ?? []), ...sourceChecks];
+  return target;
+}
