 export function parseTuple(
   inst: $ZodTuple,
   ctx: ParseContext,
   input: unknown
 ): unknown {
   if (!Array.isArray(input)) {
     return ctx.error(inst, { origin: "tuple", code: "invalid_type", input });
   }
 
   const items = inst._zod.def.items;
   const rest = inst._zod.def.rest;
 
   // check length
   const hasOptional = items.some((item) => item._zod.optin === "optional");
   const requiredCount = items.filter((item) => item._zod.optin !== "optional").length;
   const minLength = hasOptional ? requiredCount : items.length;
 
   if (rest) {
     if (input.length < minLength) {
       return ctx.error(inst, {
         origin: "tuple",
         code: "too_small",
         minimum: minLength,
         input,
       });
     }
   } else {
     if (input.length < minLength || input.length > items.length) {
       if (input.length < minLength) {
         return ctx.error(inst, {
           origin: "tuple",
           code: "too_small",
           minimum: minLength,
           input,
         });
       }
       return ctx.error(inst, {
         origin: "tuple",
         code: "too_big",
         maximum: items.length,
         input,
       });
