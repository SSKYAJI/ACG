+function _tupleMinLength(items: $ZodType[], io: "input" | "output" = "input"): number {
+  // Find the last index that is truly required (not optional, not defaulted)
+  let lastRequired = -1;
+  for (let i = 0; i < items.length; i++) {
+    const item = items[i]!;
+    const def = item._zod.def as any;
+    const isOptional = io === "input" ? item._zod.optin === "optional" : item._zod.optout === "optional";

+    if (!isOptional && !hasDefault) {
+      lastRequired = i;
+    }
