+// Utility: merge ZodObject refinements from source onto target
+// Used by ZodObject.merge() to preserve refinements like .extend() does
+export function mergeObjectRefinements<T extends { _def: { checks?: any[] } }>(
+  source: T,
+  target: T
+): T {
+  const sourceChecks = source._def.checks;
+  if (!sourceChecks || sourceChecks.length === 0) return target;
+  const targetDef = target._def as any;
+  targetDef.checks = [...(targetDef.checks ?? []), ...sourceChecks];
+  return target;
+}
