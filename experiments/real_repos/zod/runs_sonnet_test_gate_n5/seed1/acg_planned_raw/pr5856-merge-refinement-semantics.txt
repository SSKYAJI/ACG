+export function mergeObjectDefs(baseDef: Record<string, any>, incomingDef: Record<string, any>): any {
+  // Merge the two defs, with incoming taking precedence, but preserve
+  // checks (refinements) from the base def so that .merge() behaves like
+  // .extend() with respect to refinements.
+  const merged = mergeDefs(baseDef, incomingDef);
+  const baseChecks: any[] = baseDef.checks ?? [];
+  const incomingChecks: any[] = incomingDef.checks ?? [];
+  if (baseChecks.length > 0 || incomingChecks.length > 0) {
+    merged.checks = [...baseChecks, ...incomingChecks];
+  }
+  return merged;
