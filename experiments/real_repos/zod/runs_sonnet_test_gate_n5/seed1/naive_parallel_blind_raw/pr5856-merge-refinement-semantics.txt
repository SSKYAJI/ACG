+export function mergeObjects<T extends $ZodObject, U extends $ZodObject>(
+  base: T,
+  incoming: U
+): $ZodObject {
+  const baseShape = base._zod.def.shape;
+  const incomingShape = incoming._zod.def.shape;
+  const mergedShape = { ...baseShape, ...incomingShape };
+  // Preserve checks (refinements) from the base object, just like .extend() does
+  const baseChecks = base._zod.def.checks ?? [];
+  const merged = base.constructor({ ...base._zod.def, shape: mergedShape, checks: baseChecks }) as any;
+  return merged;
