+export function mergeObjects<T extends $ZodObject, U extends $ZodObject>(
+  base: T,
+  incoming: U
+): $ZodObject {
+  // extend() already preserves refinements on `base`, so we start from there.
+  const extended = base.extend(incoming.shape) as $ZodObject;
+
+  // Re-apply any refinements that were attached to `incoming` so they are not
+  // silently discarded.
+  const incomingChecks = (incoming as any)._def?.checks ?? [];
+  if (incomingChecks.length === 0) {
+    return extended;
+  }
+
+  // Clone the _def and merge the checks arrays so both sets of refinements
+  // are preserved on the resulting object.
+  const mergedDef = {
+    ...(extended as any)._def,
+    checks: [
+      ...((extended as any)._def?.checks ?? []),
+      ...incomingChecks,
+    ],
+  };
+
+  const result = new (extended.constructor as any)(mergedDef);
+  return result;
+}
