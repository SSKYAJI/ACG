   inst: $ZodTuple,
   ctx: ParseContext,
   input: unknown
 ): SyncParseResult {
   if (!Array.isArray(input)) {
     return ctx.error(inst, input, ZodIssueCode.invalid_type, {
       expected: "array",
     });
   }
 
   const items = inst._zod.def.items;
   const rest = inst._zod.def.rest;
   const issues: RawIssue[] = [];
 
   // check length
   const minLength = items.filter((item) => {
     const def = item._zod.def as any;
     return def.type !== "optional" && def.defaultValue === undefined;
   }).length;
 
   if (input.length < minLength) {
     return ctx.error(inst, input, ZodIssueCode.too_small, {
       minimum: minLength,
       inclusive: true,
       origin: "array",
     });
   }
 
   if (!rest && input.length > items.length) {
     return ctx.error(inst, input, ZodIssueCode.too_big, {
       maximum: items.length,
       inclusive: true,
       origin: "array",
     });
   }
 
   const output: unknown[] = [];
   const proms: Promise<unknown>[] = [];
 
   for (let i = 0; i < items.length; i++) {
     const item = items[i];
     const value = input[i];
     const result = item._zod.parse(value, ctx.child(inst, input, [i]));
     if (result instanceof Promise) {
       proms.push(
         result.then((r) => {
           if (r.issues) issues.push(...r.issues);
           output[i] = r.value;
         })
       );
     } else {
       if (result.issues) issues.push(...result.issues);
       output[i] = result.value;
     }
   }
 
   if (rest) {
     for (let i = items.length; i < input.length; i++) {

             output[i] = r.value;
           })
         );
       } else {
         if (result.issues) issues.push(...result.issues);
         output[i] = result.value;
       }
     }
   }
